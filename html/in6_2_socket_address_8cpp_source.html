72  void SocketAddress::setHost(const std::string& ipOrHostname) {
73  struct addrinfo hints {};
74  struct addrinfo* res;
75  struct addrinfo* resalloc;
76 
77  std::memset(&hints, 0, sizeof(hints));
78 
79  /* We only care about IPV6 results */
80  hints.ai_family = AF_INET6;
81  hints.ai_socktype = 0;
82  hints.ai_flags = AI_ADDRCONFIG | AI_V4MAPPED;
83 
84  int err = core::system::getaddrinfo(ipOrHostname.c_str(), nullptr, &hints, &res);
85 
86  if (err != 0) {
87  throw bad_hostname(ipOrHostname);
88  }
89 
90  resalloc = res;
91 
92  while (res) {
93  /* Check to make sure we have a valid AF_INET6 address */
94  if (res->ai_family == AF_INET6) {
95  sockAddr.sin6_addr = reinterpret_cast<sockaddr_in6*>(res->ai_addr)->sin6_addr;
96  break;
97  }
98 
99  res = res->ai_next;
100  }
101 
102  core::system::freeaddrinfo(resalloc);
103  }
