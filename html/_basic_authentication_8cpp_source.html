36  BasicAuthentication::BasicAuthentication(const std::string& userName, const std::string& password, const std::string& realm)
37  : userName(userName)
38  , password(password)
39  , realm(realm) {
40  std::string userNamePassword = userName + ":" + password;
41  credentials = base64::base64_encode(reinterpret_cast<unsigned char*>(userNamePassword.data()), userNamePassword.length());
42 
43  use([realm, this] MIDDLEWARE(req, res, next) {
44  std::string authCredentials = httputils::str_split(req.get("Authorization"), ' ').second;
45 
46  if (authCredentials == credentials) {
47  next();
48  } else {
49  if (authCredentials.empty()) {
50  res.set("WWW-Authenticate", "Basic realm=\"" + realm + "\"");
51  }
52  res.sendStatus(401);
53  }
54  });
55  }
