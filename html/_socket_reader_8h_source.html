70  ssize_t readFromPeer(char* junk, std::size_t junkLen) {
71  std::size_t maxReturn = std::min(junkLen, size);
72 
73  std::copy(readBuffer.data() + cursor, readBuffer.data() + cursor + maxReturn, junk);
74 
75  cursor += maxReturn;
76  size -= maxReturn;
77  ssize_t ret = static_cast<ssize_t>(maxReturn);
78 
79  return ret;

82  void doRead() {
83  errno = 0;
84 
85  if (size == 0) {
86  cursor = 0;
87 
88  std::size_t readLen = blockSize - size;
89  ssize_t retRead = read(readBuffer.data() + size, readLen);
90 
91  if (retRead > 0) {
92  size += static_cast<std::size_t>(retRead);
93 
94  if (!isSuspended()) {
95  suspend();
96  }
97  publish();
98  } else if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) {
99  if (isSuspended()) {
100  resume();
101  }
102  } else {
103  disable();
104  onError(errno);
105  }
106  } else {
107  publish();
108  }
109  }
110 
111  void shutdown() {

114  shutdownTriggered = true;
115  }
116  }

118  void terminate() override {
119  if (!terminateInProgress) {
120  setTimeout(terminateTimeout);
121  // shutdown();
122  // do not shutdown our read side because we try to do a full tcp shutdown sequence.
123  // In case we do not receive a TCP-FIN (or equivalent depending on the protocol) the
124  // connection is hard closed after "terminateTimeout".
