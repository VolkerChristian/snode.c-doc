148  void Config::prepare() {
149  parse(app["--help"]->count() > 0 || app["--help-all"]->count() > 0);
150 
151  if (app["--write-config"]->count() > 0) {
152  VLOG(0) << "Write config file " << outputConfigFile;
153  std::ofstream confFile(outputConfigFile);

155  }
156

159 
160  exit(0);
161  }
