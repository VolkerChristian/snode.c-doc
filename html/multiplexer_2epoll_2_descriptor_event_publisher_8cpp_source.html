39  void DescriptorEventPublisher::EPollEvents::muxAdd(core::DescriptorEventReceiver* eventReceiver) {
40  epoll_event ePollEvent;
41 
42  ePollEvent.data.ptr = eventReceiver;
43  ePollEvent.events = events;
44 
45  if (core::system::epoll_ctl(epfd, EPOLL_CTL_ADD, eventReceiver->getRegisteredFd(), &ePollEvent) == 0) {
46  interestCount++;
47 
48  if (interestCount >= ePollEvents.size()) {
49  ePollEvents.resize(ePollEvents.size() * 2);
50  }
51  } else if (errno == EEXIST) {
52  muxMod(eventReceiver->getRegisteredFd(), events, eventReceiver);
53  }
54  }

56  void DescriptorEventPublisher::EPollEvents::muxDel(int fd) {
57  if (core::system::epoll_ctl(epfd, EPOLL_CTL_DEL, fd, nullptr) == 0 || errno == EBADF) {
58  interestCount--;
59 
60  if (ePollEvents.size() > (interestCount * 2) + 1) {
61  ePollEvents.resize(ePollEvents.size() / 2);
