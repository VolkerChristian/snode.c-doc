50  void Daemon::daemonize(const std::string& pidFileName) {
51  if (!std::filesystem::exists(pidFileName)) {
52  pid_t pid = 0;
53 
54  /* Fork off the parent process */
55  pid = fork();
56 
57  /* An error occurred */
58  if (pid < 0) {
59  exit(EXIT_FAILURE);
60  }
61 
62  /* Success: Let the parent terminate */
63  if (pid > 0) {
64  exit(EXIT_SUCCESS);
65  }
66 
67  /* On success: The child process becomes session leader */
68  if (setsid() < 0) {
69  exit(EXIT_FAILURE);
70  }
71 
72  /* Ignore signal sent from child to parent process */
73  signal(SIGCHLD, SIG_IGN);
74 
75  /* Fork off for the second time*/
76  pid = fork();
77 
78  /* An error occurred */
79  if (pid < 0) {
80  exit(EXIT_FAILURE);
81  }
82 
83  /* Success: Let the parent terminate */
84  if (pid > 0) {
85  exit(EXIT_SUCCESS);
86  }
87 
88  /* Set new file permissions */
89  umask(0);
90 
91  /* Change the working directory to the root directory */
92  /* or another appropriated directory */
93  chdir("/");
94 
95  /* Close all open file descriptors */
96  for (long fd = sysconf(_SC_OPEN_MAX); fd >= 0; fd--) {
97  close(static_cast<int>(fd));
98  }
99 
100  /* Reopen stdin (fd = 0), stdout (fd = 1), stderr (fd = 2) */
101  if (std::freopen("/dev/null", "r", stdin) == nullptr) {
102  }
103  if (std::freopen("/dev/null", "w+", stdout) == nullptr) {
104  }
105  if (std::freopen("/dev/null", "w+", stderr) == nullptr) {
106  }
107 
108  /* Try to write PID of daemon to lockfile */
109  if (!pidFileName.empty()) {
110  std::ofstream pidFile(pidFileName, std::ofstream::out);
111 
112  if (pidFile.good()) {
113  pidFile << getpid() << std::endl;
114  pidFile.close();
115  } else {
116  exit(EXIT_FAILURE);
117  }
118  }
119  daemonized = true;
120  } else {
121  VLOG(0) << "Already running: Not daemonized ... exiting";
122  daemonized = false;
123  exit(EXIT_FAILURE);
124  }
125  }

127  void Daemon::kill(const std::string& pidFileName) {
128  /* Try to read PID of daemon to from lockfile and kill the daemon */
129  if (!pidFileName.empty()) {
130  std::ifstream pidFile(pidFileName, std::ifstream::in);
131 
132  if (pidFile.good()) {
133  int pid;
134  pidFile >> pid;
135 
136  struct pollfd pollfd;
137 
138  int pidfd = static_cast<int>(pidfd_open(pid, 0));
139 
140  if (pidfd == -1) {
141  VLOG(0) << "Daemon not running";
142  exit(EXIT_FAILURE);
143  }
144 
145  if (::kill(pid, SIGTERM) == 0) {
146  pidFile.close();
147 
148  pollfd.fd = pidfd;
149  pollfd.events = POLLIN;
150 
151  int ready = poll(&pollfd, 1, -1);
152  if (ready == -1) {
153  PLOG(ERROR) << "Poll";
154  exit(EXIT_FAILURE);
155  } else if (ready == 0) {
156  PLOG(ERROR) << "Daemon not responding";
157  exit(EXIT_FAILURE);
158  } else {
159  VLOG(0) << "Daemon terminated";
160  }
161 
162  close(pidfd);
163  exit(EXIT_SUCCESS);
164  } else {
165  PLOG(ERROR) << "Kill daemon";
166  exit(EXIT_FAILURE);
167  }
168  } else {
169  VLOG(0) << "Daemon not running";
170 
171  exit(EXIT_FAILURE);
172  }
173  }
174  }
