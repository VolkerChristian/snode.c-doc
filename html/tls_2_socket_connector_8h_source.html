53  [onConnected, this](SocketConnection* socketConnection) -> void { // onConnect
54  SSL* ssl = socketConnection->startSSL(this->ctx, 10, 2);
55 
56  if (ssl != nullptr) {
57  ssl_set_sni(ssl, this->options);
58 
59  SSL_set_connect_state(ssl);
60 
61  socketConnection->doSSLHandshake(
62  [onConnected, socketConnection](void) -> void { // onSuccess
63  LOG(INFO) << "SSL/TLS initial handshake success";
64  onConnected(socketConnection);
65  },
66  [this](void) -> void { // onTimeout
67  LOG(WARNING) << "SSL/TLS initial handshake timed out";
68  this->onError(ETIMEDOUT);
69  },
70  [this](int sslErr) -> void { // onError
71  ssl_log("SSL/TLS initial handshake failed", sslErr);
72  this->onError(-sslErr);
73  });
74  } else {
75  socketConnection->close();
76  ssl_log_error("SSL/TLS initialization failed");
77  this->onError(-SSL_ERROR_SSL);
78  }
79  },
80  [onDisconnect](SocketConnection* socketConnection) -> void { // onDisconnect
81  socketConnection->stopSSL();
82  onDisconnect(socketConnection);
83  },
84  options) {
85  ctx = ssl_ctx_new(options, false);
86  }
87
