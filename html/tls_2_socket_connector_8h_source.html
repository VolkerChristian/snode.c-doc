55  [onConnected, this](SocketConnection* socketConnection) -> void { // onConnect
56  SSL* ssl = socketConnection->startSSL(
57  this->ctx, this->clientConfig->getInitTimeout(), this->clientConfig->getShutdownTimeout());
58 
59  if (ssl != nullptr) {
60  ssl_set_sni(ssl, this->options);
61 
62  SSL_set_connect_state(ssl);
63 
64  socketConnection->doSSLHandshake(
65  [onConnected, socketConnection](void) -> void { // onSuccess
66  LOG(INFO) << "SSL/TLS initial handshake success";
67  onConnected(socketConnection);
68  },
69  [this](void) -> void { // onTimeout
70  LOG(WARNING) << "SSL/TLS initial handshake timed out";
71  this->onError(this->clientConfig->getRemoteAddress(), ETIMEDOUT);
72  },
73  [this](int sslErr) -> void { // onError
74  ssl_log("SSL/TLS initial handshake failed", sslErr);
75  this->onError(this->clientConfig->getRemoteAddress(), -sslErr);
76  });
77  } else {
78  socketConnection->close();
79  ssl_log_error("SSL/TLS initialization failed");
80  this->onError(this->clientConfig->getRemoteAddress(), -SSL_ERROR_SSL);
81  }
82  },
83  [onDisconnect](SocketConnection* socketConnection) -> void { // onDisconnect
84  socketConnection->stopSSL();
85  onDisconnect(socketConnection);
86  },
87  options) {
88  ctx = ssl_ctx_new(options, false);
89  }
90

95  void connect(const std::shared_ptr<ClientConfig>& clientConfig, const std::function<void(const SocketAddress&, int)>& onError) {
96  if (ctx == nullptr) {

99  Super::destruct();
100  } else {
101  Super::connect(clientConfig, onError);
102  }
