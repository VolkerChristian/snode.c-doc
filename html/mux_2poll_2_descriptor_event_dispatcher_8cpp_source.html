55  void DescriptorEventDispatcher::dispatchActiveEvents() {
56  pollfd* pollfds = pollFds.getEvents();
57 
58  std::unordered_map<int, PollFds::PollFdIndex> pollFdsIndices = pollFds.getPollFdIndices();
59 
60  for (auto& [fd, eventReceivers] : observedEventReceiver) { // cppcheck-suppress unassignedVariable
61  pollfd& pollFd = pollfds[pollFdsIndices[fd].index];
62 
63  if ((pollFd.events & events) != 0 && (pollFd.revents & this->revents) != 0) {
64  core::DescriptorEventReceiver* eventReceiver = eventReceivers.front();
65  if (!eventReceiver->continueImmediately() && !eventReceiver->isSuspended()) {
66  eventCounter++;
67  eventReceiver->publish();
68  }
69  }
70  }
71  }
