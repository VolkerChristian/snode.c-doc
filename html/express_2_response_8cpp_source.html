41  void Response::sendFile(const std::string& file, const std::function<void(int err)>& onError) {
42  std::string absolutFileName = file;
43 
44  if (std::filesystem::exists(absolutFileName)) {
45  std::error_code ec;
46  absolutFileName = std::filesystem::canonical(absolutFileName);
47 
48  if (std::filesystem::is_regular_file(absolutFileName, ec) && !ec) {
49  core::file::FileReader::connect(absolutFileName, *this, [this, &absolutFileName, onError](int err) -> void {
50  if (err == 0) {
51  headers.insert({{"Content-Type", web::http::MimeTypes::contentType(absolutFileName)},
52  {"Last-Modified", httputils::file_mod_http_date(absolutFileName)}});
53  headers.insert_or_assign("Content-Length", std::to_string(std::filesystem::file_size(absolutFileName)));
54  } else {
55  onError(err);
56  sendStatus(500);
57  }
58  });
59  } else {
60  errno = EACCES;
61  onError(errno);
62  sendStatus(500);
63  }
64  } else {

66  onError(errno);
