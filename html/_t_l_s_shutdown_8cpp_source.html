40  TLSShutdown::TLSShutdown(SSL* ssl,
41  const std::function<void(void)>& onSuccess,
42  const std::function<void(void)>& onTimeout,
43  const std::function<void(int err)>& onError,
44  const utils::Timeval& timeout)
45  : ReadEventReceiver("TLSShutdown", timeout)
46  , WriteEventReceiver("TLSShutdown", timeout)
47  , ssl(ssl)
48  , onSuccess(onSuccess)
49  , onTimeout(onTimeout)
50  , onError(onError)
51  , timeoutTriggered(false) {
52  fd = SSL_get_fd(ssl);
53 
54  ReadEventReceiver::enable(fd);
55  WriteEventReceiver::enable(fd);
56  ReadEventReceiver::suspend();
57  WriteEventReceiver::suspend();
58 
59  int ret = SSL_shutdown(ssl);
60 
61  int sslErr = SSL_ERROR_NONE;
62  if (ret < 0) {
63  sslErr = SSL_get_error(ssl, ret);
64  }
65 
66  switch (sslErr) {
67  case SSL_ERROR_WANT_READ:
68  ReadEventReceiver::resume();
69  break;
70  case SSL_ERROR_WANT_WRITE:
71  WriteEventReceiver::resume();
72  break;
73  case SSL_ERROR_NONE:
74  onSuccess();
75  ReadEventReceiver::disable();
76  WriteEventReceiver::disable();
77  break;
78  default:
79  onError(sslErr);
80  ReadEventReceiver::disable();
81  WriteEventReceiver::disable();
82  break;
83  }
84  }
