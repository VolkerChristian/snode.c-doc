117  void DescriptorEventDispatcher::dispatchActiveEvents(const utils::Timeval& currentTime) {
118  int count = core::system::epoll_wait(ePollEvents.getEPFd(), ePollEvents.getEvents(), ePollEvents.getInterestCount(), 0);
119 
120  for (int i = 0; i < count; i++) {
121  core::EventReceiver* eventReceiver = static_cast<core::EventReceiver*>(ePollEvents.getEvents()[i].data.ptr);
122  if (!eventReceiver->continueImmediately() && !eventReceiver->isSuspended()) {
123  eventCounter++;
124  eventReceiver->dispatch(currentTime);
125  }
126  }
