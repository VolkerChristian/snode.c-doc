79  TickStatus EventLoop::_tick(struct timeval timeOut) {
80  TickStatus tickStatus = TickStatus::SUCCESS;
81 
82  struct timeval nextDispatcherTimeout = readEventDispatcher.observeEnabledEvents();
83  nextEventTimeout = std::min(nextDispatcherTimeout, nextEventTimeout);
84 
85  nextDispatcherTimeout = writeEventDispatcher.observeEnabledEvents();
86  nextEventTimeout = std::min(nextDispatcherTimeout, nextEventTimeout);
87 
88  nextDispatcherTimeout = exceptionalConditionEventDispatcher.observeEnabledEvents();
89  nextEventTimeout = std::min(nextDispatcherTimeout, nextEventTimeout);
90 
91  nextDispatcherTimeout = timerEventDispatcher.getNextTimeout();
92  nextEventTimeout = std::min(nextDispatcherTimeout, nextEventTimeout);
93 
94  int maxFd = readEventDispatcher.getMaxFd();
95  maxFd = std::max(writeEventDispatcher.getMaxFd(), maxFd);
96  maxFd = std::max(exceptionalConditionEventDispatcher.getMaxFd(), maxFd);
97 
98  if (maxFd >= 0 || !timerEventDispatcher.empty()) {
99  nextEventTimeout = std::max(nextEventTimeout, {0, 0});
100  nextEventTimeout = std::min(nextEventTimeout, timeOut);
101 
102  int counter = core::system::select(maxFd + 1,
103  &readEventDispatcher.getFdSet(),
104  &writeEventDispatcher.getFdSet(),
105  &exceptionalConditionEventDispatcher.getFdSet(),
106  &nextEventTimeout);
107 
108  if (counter >= 0) {
109  nextEventTimeout = {LONG_MAX, 0};
110 
111  timerEventDispatcher.dispatch();
112 
113  struct timeval currentTime = {core::system::time(nullptr), 0};
114 
115  nextDispatcherTimeout = readEventDispatcher.dispatchActiveEvents(currentTime);
116  nextEventTimeout = std::min(nextDispatcherTimeout, nextEventTimeout);
117 
118  nextDispatcherTimeout = writeEventDispatcher.dispatchActiveEvents(currentTime);
119  nextEventTimeout = std::min(nextDispatcherTimeout, nextEventTimeout);
120 
121  nextDispatcherTimeout = exceptionalConditionEventDispatcher.dispatchActiveEvents(currentTime);
122  nextEventTimeout = std::min(nextDispatcherTimeout, nextEventTimeout);
123 
124  readEventDispatcher.unobserveDisabledEvents();
125  writeEventDispatcher.unobserveDisabledEvents();
126  exceptionalConditionEventDispatcher.unobserveDisabledEvents();
127 
128  readEventDispatcher.releaseUnobservedEvents();
129  writeEventDispatcher.releaseUnobservedEvents();
130  exceptionalConditionEventDispatcher.releaseUnobservedEvents();
131 
132  DynamicLoader::execDlCloseDeleyed();
133  } else if (errno != EINTR) {
134  PLOG(ERROR) << "select";
135  tickStatus = TickStatus::SELECT_ERROR;
136  }
137  } else {
138  tickStatus = TickStatus::NO_OBSERVER;
139  }
140 
141  return tickStatus;
142  }

175  int EventLoop::start(struct timeval timeOut) {
176  if (!initialized) {
177  PLOG(ERROR) << "snode.c not initialized. Use EventLoop::init(argc, argv) before EventLoop::start().";
178  exit(1);
179  }
180 
181  sighandler_t oldSigPipeHandler = core::system::signal(SIGPIPE, SIG_IGN);
182  sighandler_t oldSigQuitHandler = core::system::signal(SIGQUIT, EventLoop::stoponsig);
183  sighandler_t oldSigHubHandler = core::system::signal(SIGHUP, EventLoop::stoponsig);
184  sighandler_t oldSigIntHandler = core::system::signal(SIGINT, EventLoop::stoponsig);
185  sighandler_t oldSigTermHandler = core::system::signal(SIGTERM, EventLoop::stoponsig);
186  sighandler_t oldSigAbrtHandler = core::system::signal(SIGABRT, EventLoop::stoponsig);
187 
188  if (!running) {
189  running = true;
190 
191  stopped = false;
192 
193  do {
194  while (!stopped && eventLoop._tick(timeOut) == TickStatus::SUCCESS) {
195  eventLoop.tickCounter++;
196  };
197 
198  eventLoop._free();
199  } while (!stopped && false);
200 
201  running = false;
202  }
203 
204  core::system::signal(SIGPIPE, oldSigPipeHandler);
205  core::system::signal(SIGQUIT, oldSigQuitHandler);
206  core::system::signal(SIGHUP, oldSigHubHandler);
207  core::system::signal(SIGINT, oldSigIntHandler);
208  core::system::signal(SIGTERM, oldSigTermHandler);
209  core::system::signal(SIGABRT, oldSigAbrtHandler);
210 
211  int returnReason = 0;
212 
213  if (stopsig != 0) {
214  returnReason = -1;
215  }
216 
217  return returnReason;
218  }
