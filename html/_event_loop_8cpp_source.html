79  TickStatus EventLoop::_tick(struct timeval timeOut) {
80  TickStatus tickStatus = TickStatus::SUCCESS;
81 
82  struct timeval nextTimeout = readEventDispatcher.observeEnabledEvents();
83  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
84 
85  nextTimeout = writeEventDispatcher.observeEnabledEvents();
86  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
87 
88  nextTimeout = exceptionalConditionEventDispatcher.observeEnabledEvents();
89  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
90 
91  nextTimeout = timerEventDispatcher.getNextTimeout();
92  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
93 
94  int maxFd = readEventDispatcher.getMaxFd();
95  maxFd = std::max(writeEventDispatcher.getMaxFd(), maxFd);
96  maxFd = std::max(exceptionalConditionEventDispatcher.getMaxFd(), maxFd);
97 
98  if (maxFd >= 0 || !timerEventDispatcher.empty()) {
99  nextInactivityTimeout = std::max(nextInactivityTimeout, {0, 0});
100  nextInactivityTimeout = std::min(nextInactivityTimeout, timeOut);
101 
102  int counter = net::system::select(maxFd + 1,
103  &readEventDispatcher.getFdSet(),
104  &writeEventDispatcher.getFdSet(),
105  &exceptionalConditionEventDispatcher.getFdSet(),
106  &nextInactivityTimeout);
107 
108  if (counter >= 0) {
109  nextInactivityTimeout = {LONG_MAX, 0};
110 
111  timerEventDispatcher.dispatch();
112 
113  struct timeval currentTime = {net::system::time(nullptr), 0};
114 
115  nextTimeout = readEventDispatcher.dispatchActiveEvents(currentTime);
116  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
117 
118  nextTimeout = writeEventDispatcher.dispatchActiveEvents(currentTime);
119  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
120 
121  nextTimeout = exceptionalConditionEventDispatcher.dispatchActiveEvents(currentTime);
122  nextInactivityTimeout = std::min(nextTimeout, nextInactivityTimeout);
123 
124  readEventDispatcher.unobserveDisabledEvents();
125  writeEventDispatcher.unobserveDisabledEvents();
126  exceptionalConditionEventDispatcher.unobserveDisabledEvents();
127 
128  readEventDispatcher.releaseUnobservedEvents();
129  writeEventDispatcher.releaseUnobservedEvents();
130  exceptionalConditionEventDispatcher.releaseUnobservedEvents();
131 
132  DynamicLoader::execDlCloseDeleyed();
133  } else if (errno != EINTR) {
134  PLOG(ERROR) << "select";
135  tickStatus = TickStatus::SELECT_ERROR;
136  }
137  } else {
138  tickStatus = TickStatus::NO_OBSERVER;
139  }
140 
141  return tickStatus;
142  }

175  int EventLoop::start(struct timeval timeOut) {
176  if (!initialized) {
177  PLOG(ERROR) << "snode.c not initialized. Use EventLoop::init(argc, argv) before EventLoop::start().";
178  exit(1);
179  }
180 
181  sighandler_t oldSigPipeHandler = net::system::signal(SIGPIPE, SIG_IGN);
182  sighandler_t oldSigQuitHandler = net::system::signal(SIGQUIT, EventLoop::stoponsig);
183  sighandler_t oldSigHubHandler = net::system::signal(SIGHUP, EventLoop::stoponsig);
184  sighandler_t oldSigIntHandler = net::system::signal(SIGINT, EventLoop::stoponsig);
185  sighandler_t oldSigTermHandler = net::system::signal(SIGTERM, EventLoop::stoponsig);
186  sighandler_t oldSigAbrtHandler = net::system::signal(SIGABRT, EventLoop::stoponsig);
187 
188  if (!running) {
189  running = true;
190 
191  stopped = false;
192 
193  do {
194  while (!stopped && eventLoop._tick(timeOut) == TickStatus::SUCCESS) {
195  eventLoop.tickCounter++;
196  };
197 
198  eventLoop._free();
199  } while (!stopped && false);
200 
201  running = false;
202  }
203 
204  net::system::signal(SIGPIPE, oldSigPipeHandler);
205  net::system::signal(SIGQUIT, oldSigQuitHandler);
206  net::system::signal(SIGHUP, oldSigHubHandler);
207  net::system::signal(SIGINT, oldSigIntHandler);
208  net::system::signal(SIGTERM, oldSigTermHandler);
209  net::system::signal(SIGABRT, oldSigAbrtHandler);
210 
211  int returnReason = 0;
212 
213  if (stopsig != 0) {
214  returnReason = -1;
215  }
216 
217  return returnReason;
218  }
