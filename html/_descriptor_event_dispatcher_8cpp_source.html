40  void DescriptorEventDispatcher::enable(core::DescriptorEventReceiver* eventReceiver) {
41  int fd = eventReceiver->getRegisteredFd();
42 
43  if (disabledEventReceiver.contains(fd) && disabledEventReceiver[fd].contains(eventReceiver)) {
44  // same tick as disable
45  disabledEventReceiver[fd].remove(eventReceiver);
46  } else if (!eventReceiver->isEnabled() &&
47  (!enabledEventReceiver.contains(fd) || !enabledEventReceiver[fd].contains(eventReceiver))) {
48  // next tick as disable
49  enabledEventReceiver[fd].push_back(eventReceiver);
50  } else {
51  LOG(WARNING) << "EventReceiver double enable " << fd;
52  }

97  void DescriptorEventDispatcher::observeEnabledEvents(const utils::Timeval& currentTime) {
98  for (const auto& [fd, eventReceivers] : enabledEventReceiver) { // cppcheck-suppress unassignedVariable
99  for (core::DescriptorEventReceiver* eventReceiver : eventReceivers) {
100  if (eventReceiver->isEnabled()) {
101  eventReceiver->triggered(currentTime);
102  observedEventReceiver[fd].push_front(eventReceiver);
103  modAdd(eventReceiver);
104  if (eventReceiver->isSuspended()) {
105  modOff(eventReceiver);

131  void DescriptorEventDispatcher::unobserveDisabledEvents([[maybe_unused]] const utils::Timeval& currentTime) {
132  std::map<int, EventReceiverList> unobservedEventReceiver;
133 
134  for (const auto& [fd, eventReceivers] : disabledEventReceiver) {
135  for (core::DescriptorEventReceiver* eventReceiver : eventReceivers) {
136  observedEventReceiver[fd].remove(eventReceiver);
137  if (observedEventReceiver[fd].empty()) {
138  modDel(eventReceiver);
139  observedEventReceiver.erase(fd);
140  } else {
141  observedEventReceiver[fd].front()->triggered(currentTime);
142  if (!observedEventReceiver[fd].front()->isSuspended()) {
143  modOn(observedEventReceiver[fd].front());
144  } else {
145  modOff(observedEventReceiver[fd].front());
146  }
147  }
148  eventReceiver->disabled();
149  if (eventReceiver->getObservationCounter() == 0) {
150  unobservedEventReceiver[fd].push_back(eventReceiver);
151  }
152  }
153  }
154 
155  disabledEventReceiver.clear();

158  for (const auto& [fd, eventReceivers] : unobservedEventReceiver) { // cppcheck-suppress unusedVariable
159  for (DescriptorEventReceiver* eventReceiver : eventReceivers) {
160  eventReceiver->unobservedEvent();
161  }
162  }

166  }

186  utils::Timeval DescriptorEventDispatcher::getNextTimeout(const utils::Timeval& currentTime) const {
187  utils::Timeval nextTimeout = core::DescriptorEventReceiver::TIMEOUT::MAX;
188 
189  for (const auto& [fd, eventReceivers] : observedEventReceiver) { // cppcheck-suppress unusedVariable
190  const core::DescriptorEventReceiver* eventReceiver = eventReceivers.front();
191 
192  if (eventReceiver->isEnabled()) {
193  if (!eventReceiver->isSuspended() && eventReceiver->continueImmediately()) {
194  nextTimeout = 0;
195  } else {
196  nextTimeout = std::min(eventReceiver->getTimeout(currentTime), nextTimeout);
197  }
198  } else {
199  nextTimeout = 0;
200  }
201  }
202
