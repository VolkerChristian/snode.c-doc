40  void DescriptorEventDispatcher::enable(core::EventReceiver* eventReceiver) {
41  int fd = eventReceiver->getRegisteredFd();
42 
43  if (disabledEventReceiver.contains(fd) && disabledEventReceiver[fd].contains(eventReceiver)) {
44  // same tick as disable
45  disabledEventReceiver[fd].remove(eventReceiver);
46  } else if (!eventReceiver->isEnabled() &&
47  (!enabledEventReceiver.contains(fd) || !enabledEventReceiver[fd].contains(eventReceiver))) {
48  // next tick as disable
49  enabledEventReceiver[fd].push_back(eventReceiver);
50  } else {
51  LOG(WARNING) << "EventReceiver double enable " << fd;

126  void DescriptorEventDispatcher::unobserveDisabledEvents(const utils::Timeval& currentTime) {
127  for (const auto& [fd, eventReceivers] : disabledEventReceiver) {
128  for (core::EventReceiver* eventReceiver : eventReceivers) {
129  observedEventReceiver[fd].remove(eventReceiver);
130  if (observedEventReceiver[fd].empty()) {
131  modDel(eventReceiver);
132  observedEventReceiver.erase(fd);
133  } else if (!observedEventReceiver[fd].front()->isSuspended()) {
134  modOn(observedEventReceiver[fd].front());
135  observedEventReceiver[fd].front()->triggered(currentTime);
136  } else {
137  modOff(observedEventReceiver[fd].front());
138  }
139  eventReceiver->disabled();
140  if (eventReceiver->getObservationCounter() == 0) {
141  unobservedEventReceiver[fd].push_back(eventReceiver);
142  }
143  }
144  }

146  disabledEventReceiver.clear();

177  utils::Timeval DescriptorEventDispatcher::getNextTimeout(const utils::Timeval& currentTime) const {
178  utils::Timeval nextTimeout = core::EventReceiver::TIMEOUT::MAX;
179 
180  for (const auto& [fd, eventReceivers] : observedEventReceiver) { // cppcheck-suppress unusedVariable
181  const core::EventReceiver* eventReceiver = eventReceivers.front();
182 
183  if (eventReceiver->isEnabled()) {
184  if (!eventReceiver->isSuspended() && eventReceiver->continueImmediately()) {
185  nextTimeout = 0;
186  } else {
187  nextTimeout = std::min(eventReceiver->getTimeout(currentTime), nextTimeout);
188  }
189  } else {
190  nextTimeout = 0;
191  }
192  }
193
