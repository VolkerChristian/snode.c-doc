74  void listen(const SocketAddress& bindAddress, int backlog, const std::function<void(int)>& onError) {
75  Socket::open([this, &bindAddress, &backlog, &onError](int errnum) -> void {
76  if (errnum > 0) {
77  onError(errnum);
78  destruct();
79  } else {
80 #if !defined(NDEBUG)
81  reuseAddress([this, &bindAddress, &backlog, &onError](int errnum) -> void {
82  if (errnum != 0) {
83  onError(errnum);
84  destruct();
85  } else {
86 #endif
87  Socket::bind(bindAddress, [this, &backlog, &onError](int errnum) -> void {
88  if (errnum > 0) {
89  onError(errnum);

122  void acceptEvent() override {
123  typename SocketAddress::SockAddr remoteAddress{};
124  socklen_t remoteAddressLength = sizeof(remoteAddress);
125 
126  int fd = -1;
127 
128  fd = net::system::accept4(
129  SocketAcceptor::getFd(), reinterpret_cast<struct sockaddr*>(&remoteAddress), &remoteAddressLength, SOCK_NONBLOCK);
130 
131  if (fd >= 0) {
132  typename SocketAddress::SockAddr localAddress{};
133  socklen_t addressLength = sizeof(localAddress);
134 
135  if (net::system::getsockname(fd, reinterpret_cast<sockaddr*>(&localAddress), &addressLength) == 0) {
136  SocketConnection* socketConnection = new SocketConnection(
137  fd, socketContextFactory, SocketAddress(localAddress), SocketAddress(remoteAddress), onConnect, onDisconnect);
138 
139  onConnected(socketConnection);
140  } else {
141  PLOG(ERROR) << "getsockname";
142  net::system::shutdown(fd, SHUT_RDWR);
143  net::system::close(fd);
144  }
145  } else if (errno != EINTR) {
146  PLOG(ERROR) << "accept";
147  }
148  }
