85  void listen(const SocketAddress& bindAddress, int backlog, const std::function<void(const Socket& socket, int)>& onError) {
86  Socket::open(
87  [this, &bindAddress, &backlog, &onError](int errnum) -> void {
88  if (errnum > 0) {
89  onError(static_cast<const Socket&>(*this), errnum);
90  destruct();
91  } else {
92 #if !defined(NDEBUG)
93  reuseAddress([this, &bindAddress, &backlog, &onError](int errnum) -> void {
94  if (errnum != 0) {
95  onError(static_cast<const Socket&>(*this), errnum);
96  destruct();
97  } else {
98 #endif
99  Socket::bind(bindAddress, [this, &backlog, &onError](int errnum) -> void {
100  if (errnum > 0) {
101  onError(static_cast<const Socket&>(*this), errnum);
102  destruct();
103  } else {
104  int ret = core::system::listen(Socket::fd, backlog);
105 
106  if (ret == 0) {
107  enable(Socket::fd);
108  onError(static_cast<const Socket&>(*this), 0);
109  } else {
110  onError(static_cast<const Socket&>(*this), errno);
111  destruct();
112  }
113  }
114  });
115 #if !defined(NDEBUG)
116  }
117  });
118 #endif
119  }
120  },
121  SOCK_NONBLOCK);
122  }

135  void acceptEvent() override {
136  typename SocketAddress::SockAddr remoteAddress{};
137  socklen_t remoteAddressLength = sizeof(remoteAddress);
138 
139  int fd = -1;
140 
141  int count = MAX_ACCEPTS_PER_TICK;
142 
143  do {
144  fd = core::system::accept4(
145  Socket::fd, reinterpret_cast<struct sockaddr*>(&remoteAddress), &remoteAddressLength, SOCK_NONBLOCK);
146 
147  if (fd >= 0) {
148  typename SocketAddress::SockAddr localAddress{};
149  socklen_t addressLength = sizeof(localAddress);
150 
151  if (core::system::getsockname(fd, reinterpret_cast<sockaddr*>(&localAddress), &addressLength) == 0) {
152  SocketConnection* socketConnection = new SocketConnection(fd,
153  socketContextFactory,
154  SocketAddress(localAddress),
155  SocketAddress(remoteAddress),
156  onConnect,
157  onDisconnect,
158  serverConfig.getReadTimeout(),
159  serverConfig.getWriteTimeout(),
160  serverConfig.getReadBlockSize(),
161  serverConfig.getWriteBlockSize(),
162  serverConfig.getTerminateTimeout());
163 
164  onConnected(socketConnection);
165  } else {
166  PLOG(ERROR) << "getsockname";
167  core::system::shutdown(fd, SHUT_RDWR);
168  core::system::close(fd);
169  }
170  }
171  } while (fd >= 0 && --count > 0);
172 
173  if (fd < 0 && errno != EINTR && errno != EAGAIN) {
174  PLOG(ERROR) << "accept";
175  }
176  }
