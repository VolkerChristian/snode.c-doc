155  void acceptEvent() override {
156  if (config->getClusterMode() == net::config::ConfigCluster::MODE::NONE ||
157  config->getClusterMode() == net::config::ConfigCluster::MODE::PRIMARY) {
158  PrimarySocket socket;
159 
160  int acceptsPerTick = config->getAcceptsPerTick();
161 
162  do {
163  SocketAddress remoteAddress{};
164  socket = primarySocket->accept4(remoteAddress, SOCK_NONBLOCK);
165  if (socket.isValid()) {
166  if (config->getClusterMode() == net::config::ConfigCluster::MODE::NONE) {
167  socketConnectionFactory.create(socket, config);
168  } else {
169  // Send descriptor to SECONDARY
170  VLOG(0) << "Sending to secondary";
171  secondarySocket->sendFd(SecondarySocket::SocketAddress("/tmp/secondary"), socket.getFd());
172  SecondarySocket::SocketAddress address;
173  }
174  } else if (errno != EINTR && errno != EAGAIN && errno != EWOULDBLOCK) {
175  PLOG(ERROR) << "accept";
176  }
177  } while (--acceptsPerTick > 0 && socket.isValid());
178  } else if (config->getClusterMode() == net::config::ConfigCluster::MODE::SECONDARY ||
179  config->getClusterMode() == net::config::ConfigCluster::MODE::PROXY) {
180  // Receive socketfd via SOCK_UNIX, SOCK_DGRAM
181  int fd = -1;
182 
183  if (secondarySocket->recvFd(&fd) >= 0) {
184  PrimarySocket socket(fd);
185 
186  if (config->getClusterMode() == net::config::ConfigCluster::MODE::SECONDARY) {
187  socketConnectionFactory.create(socket, config);
188  } else { // PROXY
189  // Send to SECONDARY (TERTIARY)
190  }
191  } else {
192  PLOG(ERROR) << "read_fd";
193  }
194  }
