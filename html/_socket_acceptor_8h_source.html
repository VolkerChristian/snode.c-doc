143  void acceptEvent() override {
144  typename SocketAddress::SockAddr remoteAddress{};
145  socklen_t remoteAddressLength = sizeof(remoteAddress);
146 
147  int fd = -1;
148 
149  int acceptsPerTick = config->getAcceptsPerTick();
150 
151  do {
152  fd = core::system::accept4(
153  Socket::getFd(), reinterpret_cast<struct sockaddr*>(&remoteAddress), &remoteAddressLength, SOCK_NONBLOCK);
154 
155  if (fd >= 0) {
156  typename SocketAddress::SockAddr localAddress{};
157  socklen_t addressLength = sizeof(localAddress);
158 
159  if (core::system::getsockname(fd, reinterpret_cast<sockaddr*>(&localAddress), &addressLength) == 0) {
160  SocketConnection* socketConnection = new SocketConnection(fd,
161  socketContextFactory,
162  SocketAddress(localAddress),
163  SocketAddress(remoteAddress),
164  onConnect,
165  onDisconnect,
166  config->getReadTimeout(),
167  config->getWriteTimeout(),
168  config->getReadBlockSize(),
169  config->getWriteBlockSize(),
170  config->getTerminateTimeout());
171 
172  onConnected(socketConnection);
173  } else {
174  PLOG(ERROR) << "getsockname";
175  core::system::shutdown(fd, SHUT_RDWR);
176  core::system::close(fd);
177  }
178  }
179  } while (fd >= 0 && --acceptsPerTick > 0);
180 
181  if (fd < 0 && errno != EINTR && errno != EAGAIN) {
182  PLOG(ERROR) << "accept";
183  }
184  }
