69  void Logger::setLogLevel(int level) {
70  conf.set(el::Level::Trace, el::ConfigurationType::Enabled, "false"); // trace method/function calling
71  conf.set(el::Level::Debug, el::ConfigurationType::Enabled, "false"); // typical assert messages - but we can go on
72  conf.set(el::Level::Info, el::ConfigurationType::Enabled, "false"); // additional infos - what 's going on
73  conf.set(el::Level::Warning, el::ConfigurationType::Enabled, "false"); // not that serious but mentioning
74  conf.set(el::Level::Error, el::ConfigurationType::Enabled, "false"); // serious errors - but we can go on
75  conf.set(el::Level::Fatal, el::ConfigurationType::Enabled, "false"); // asserts - stop - we can not go on
76 
77  switch (level) {

79  conf.set(el::Level::Trace, el::ConfigurationType::Enabled, "true");

83  [[fallthrough]];

86  [[fallthrough]];
87  case 3:
88  conf.set(el::Level::Warning, el::ConfigurationType::Enabled, "true");
89  [[fallthrough]];
