37  MariaDBConnection::MariaDBConnection(MariaDBClient* mariaDBClient, const MariaDBConnectionDetails& connectionDetails)
38  : ReadEventReceiver("MariaDBConnectionRead", core::DescriptorEventReceiver::TIMEOUT::DISABLE)
39  , WriteEventReceiver("MariaDBConnectionWrite", core::DescriptorEventReceiver::TIMEOUT::DISABLE)
40  , ExceptionalConditionEventReceiver("MariaDBConnectionExceptional", core::DescriptorEventReceiver::TIMEOUT::DISABLE)
41  , mariaDBClient(mariaDBClient)
42  , mysql(mysql_init(nullptr))
43  , commandStartEvent("MariaDBCommandStartEvent", this) {
44  mysql_options(mysql, MYSQL_OPT_NONBLOCK, nullptr);
45 
46  execute_async(std::move(MariaDBCommandSequence().execute_async(new database::mariadb::commands::async::MariaDBConnectCommand(
47  connectionDetails,
48  [this](void) -> void {
49  if (mysql_errno(mysql) == 0) {
50  int fd = mysql_get_socket(mysql);
51 
52  VLOG(0) << "Got valid descriptor: " << fd;
53 
54  ReadEventReceiver::enable(fd);
55  WriteEventReceiver::enable(fd);
56  ExceptionalConditionEventReceiver::enable(fd);
57 
58  ReadEventReceiver::suspend();
59  WriteEventReceiver::suspend();
60  ExceptionalConditionEventReceiver::suspend();
61 
62  connected = true;
63  } else {
64  VLOG(0) << "Got no valid descriptor: " << mysql_error(mysql) << ", " << mysql_errno(mysql);
65  }
66  },
67  [](void) -> void {
68  VLOG(0) << "Connect success";
69  },
70  [](const std::string& errorString, unsigned int errorNumber) -> void {
71  VLOG(0) << "Connect error: " << errorString << " : " << errorNumber;
72  }))));
73  }

115  void MariaDBConnection::commandStart(const utils::Timeval& currentTime) {
116  if (!commandSequenceQueue.empty()) {
117  currentCommand = commandSequenceQueue.front().nextCommand();
118  currentCommand->setMariaDBConnection(this);
119 
120  // VLOG(0) << "Start: " << currentCommand->commandInfo();
121 
122  int status = currentCommand->commandStart(mysql, currentTime);
123  checkStatus(status);
124  } else if (mariaDBClient != nullptr) {
125  if (ReadEventReceiver::isSuspended()) {
126  ReadEventReceiver::resume();
127  }
128  } else {
129  ReadEventReceiver::disable();
130  WriteEventReceiver::disable();
131  ExceptionalConditionEventReceiver::disable();
132  }
133  }

135  void MariaDBConnection::commandContinue(int status) {

165  void MariaDBConnection::checkStatus(int status) {
166  if (connected) {
167  if ((status & MYSQL_WAIT_READ) != 0) {
168  if (ReadEventReceiver::isSuspended()) {
169  ReadEventReceiver::resume();
170  }
171  } else if (!ReadEventReceiver::isSuspended()) {
172  ReadEventReceiver::suspend();
173  }
174 
175  if ((status & MYSQL_WAIT_WRITE) != 0) {
176  if (WriteEventReceiver::isSuspended()) {
177  WriteEventReceiver::resume();
178  }
179  } else if (!WriteEventReceiver::isSuspended()) {
180  WriteEventReceiver::suspend();
181  }
182 
183  if ((status & MYSQL_WAIT_EXCEPT) != 0) {
184  if (ExceptionalConditionEventReceiver::isSuspended()) {
185  ExceptionalConditionEventReceiver::resume();
186  }
187  } else if (!ExceptionalConditionEventReceiver::isSuspended()) {
188  ExceptionalConditionEventReceiver::suspend();
189  }
190 
191  if ((status & MYSQL_WAIT_TIMEOUT) != 0) {
192  ReadEventReceiver::setTimeout(mysql_get_timeout_value(mysql));
193  // WriteEventReceiver::setTimeout(mysql_get_timeout_value(mysql));
194  // ExceptionalConditionEventReceiver::setTimeout(mysql_get_timeout_value(mysql));
195  } else {
196  ReadEventReceiver::setTimeout(core::DescriptorEventReceiver::TIMEOUT::DEFAULT);
197  // WriteEventReceiver::setTimeout(core::DescriptorEventReceiver::TIMEOUT::DEFAULT);
198  // ExceptionalConditionEventReceiver::setTimeout(core::DescriptorEventReceiver::TIMEOUT::DEFAULT);
199  }
200 
201  if (status == 0) {
202  if (mysql_errno(mysql) == 0) {
203  if (currentCommand->commandCompleted()) {
204  commandCompleted();
205  }
206  } else {
207  currentCommand->commandError(mysql_error(mysql), mysql_errno(mysql));
208  commandCompleted();
209  }
210  commandStartEvent.publish();
211  }
212  } else {
213  currentCommand->commandError(mysql_error(mysql), mysql_errno(mysql));
214  commandCompleted();
215  delete this;
216  }
217  }
