86  TickStatus EventMultiplexer::tick(const utils::Timeval& tickTimeOut) {
87  TickStatus tickStatus = TickStatus::SUCCESS;
88 
89  utils::Timeval currentTime = utils::Timeval::currentTime();
90 
91  publishActiveEvents(currentTime);
92  executeEventQueue(currentTime);
93  checkTimedOutEvents(currentTime);
94  purgeReleasedResources(currentTime);
95 
96  if (getObservedEventReceiverCount() > 0 || !timerEventPublisher->empty() || !eventQueue.empty()) {
97  utils::Timeval nextTimeout = std::min(getNextTimeout(currentTime), tickTimeOut);
98 
99  activeEventCount = multiplex(nextTimeout);
100 
101  if (activeEventCount < 0 && errno != EINTR) {
102  tickStatus = TickStatus::ERROR;
103  }
104  } else {
105  tickStatus = TickStatus::NO_OBSERVER;
106  }
107 
108  return tickStatus;
109  }

118  utils::Timeval EventMultiplexer::getNextTimeout(const utils::Timeval& currentTime) {
119  utils::Timeval nextTimeout = DescriptorEventReceiver::TIMEOUT::MAX;
120 
121  if (eventQueue.empty()) {
122  for (DescriptorEventPublisher* const eventMultiplexer : descriptorEventPublishers) {
123  nextTimeout = std::min(eventMultiplexer->getNextTimeout(currentTime), nextTimeout);
124  }
125  nextTimeout = std::min(timerEventPublisher->getNextTimeout(currentTime), nextTimeout);
126  nextTimeout = std::max(nextTimeout, utils::Timeval()); // In case nextTimeout is negative
127  } else {
128  nextTimeout = 0;
129  }
130 
131  return nextTimeout;
132  }

180  void EventMultiplexer::EventQueue::execute(const utils::Timeval& currentTime) {
181  std::swap(executeQueue, publishQueue);
182 
183  sigprocmask(SIG_BLOCK, &newSet, &oldSet);
184 
185  for (Event* event : *executeQueue) {
186  event->dispatch(currentTime);
187  }
188 
189  sigprocmask(SIG_SETMASK, &oldSet, &newSet);
190 
191  executeQueue->clear();
192  }
