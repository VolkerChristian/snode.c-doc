86  TickStatus EventMultiplexer::tick(const utils::Timeval& tickTimeOut, bool stopped) {
87  TickStatus tickStatus = TickStatus::SUCCESS;
88 
89  utils::Timeval currentTime = utils::Timeval::currentTime();
90 
91  eventQueue.execute(currentTime);
92 
93  unobserveDisabledEvents(currentTime);
94  checkTimedOutEvents(currentTime);
95 
96  DynamicLoader::execDlCloseDeleyed();
97 
98  if (getObservedEventReceiverCount() > 0 || (!timerEventPublisher->empty() && !stopped)) {
99  utils::Timeval nextTimeout = 0;
100 
101  if (eventQueue.empty()) {
102  nextTimeout = getNextTimeout(currentTime);
103 
104  nextTimeout = std::min(nextTimeout, tickTimeOut);
105  nextTimeout = std::max(nextTimeout, utils::Timeval()); // In case nextEventTimeout is negativ
106  }
107 
108  int ret = multiplex(nextTimeout);
109 
110  if (ret >= 0) {
111  currentTime = utils::Timeval::currentTime();
112  dispatchActiveEvents(ret, currentTime);
113  } else if (errno != EINTR) {
114  tickStatus = TickStatus::ERROR;
115  } else {
116  // ignore EINTR - it is no error
117  }
118  } else {
119  tickStatus = TickStatus::NO_OBSERVER;
120  }
121 
122  return tickStatus;
123  }

125  void EventMultiplexer::stop() {
126  core::TickStatus tickStatus;
127 
128  do {
129  for (DescriptorEventPublisher* const eventMultiplexer : descriptorEventPublishers) {
130  eventMultiplexer->stop();
131  }
132  tickStatus = tick(2, true);
133  } while (tickStatus == TickStatus::SUCCESS);
134 
135  do {
136  timerEventPublisher->stop();
137  tickStatus = tick(0, false);
138  } while (tickStatus == TickStatus::SUCCESS);
