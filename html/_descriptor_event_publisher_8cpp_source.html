65  void DescriptorEventPublisher::unobserveDisabledEvents(const utils::Timeval& currentTime) {
66  if (observedEventReceiversDirty) {
67  std::erase_if(observedEventReceivers, [this, &currentTime](auto& observedEventReceiversEntry) -> bool {
68  auto& [fdTmp, observedEventReceiverList] = observedEventReceiversEntry; // cppcheck-suppress constVariable
69  int fd = fdTmp; // NOTE: Needed because clang did not capture compound initialized variables. Bug in clang?
70  std::erase_if(observedEventReceiverList, [fd](DescriptorEventReceiver* descriptorEventReceiver) -> bool {
71  bool isDisabled = !descriptorEventReceiver->isEnabled();
72  if (isDisabled) {
73  descriptorEventReceiver->setDisabled();
74  if (!descriptorEventReceiver->isObserved()) {
75  LOG(INFO) << "UnobservedEvent: " << descriptorEventReceiver->getName() << ", fd = " << fd;
76  descriptorEventReceiver->unobservedEvent();
77  } else {
78  LOG(INFO) << "Unobserved: " << descriptorEventReceiver->getName() << ", fd = " << fd;
79  }
80  }
81  return isDisabled;
82  });
83  if (observedEventReceiverList.empty()) {
84  muxDel(fd);
85  } else {
86  DescriptorEventReceiver* descriptorEventReceiver = observedEventReceivers[fd].front();
87  descriptorEventReceiver->triggered(currentTime);
88  if (!descriptorEventReceiver->isSuspended()) {
89  muxOn(descriptorEventReceiver);
90  } else {
91  muxOff(fd);
92  }
93  }
94  return observedEventReceiverList.empty();
95  });
96  observedEventReceiversDirty = false;
97  }
98  }
