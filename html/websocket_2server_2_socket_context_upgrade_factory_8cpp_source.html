39  SocketContextUpgrade* SocketContextUpgradeFactory::create(core::socket::SocketConnection* socketConnection,
40  web::http::server::Request* request,
41  web::http::server::Response* response) {
42  SocketContextUpgrade* socketContext = nullptr;
43 
44  if (request->header("Sec-WebSocket-Version") == "13") {
45  std::string subProtocolNames = request->header("sec-websocket-protocol");
46  std::string subProtocolName;
47 
48  do {
49  std::tie(subProtocolName, subProtocolNames) = httputils::str_split(subProtocolNames, ',');
50  httputils::str_trimm(subProtocolName);
51 
52  socketContext = SocketContextUpgrade::create(this, socketConnection, subProtocolName);
53  } while (subProtocolNames.length() > 0 && socketContext == nullptr);
54 
55  if (socketContext != nullptr) {
56  response->set("Upgrade", "websocket");
57  response->set("Connection", "Upgrade");
58  response->set("Sec-WebSocket-Protocol", subProtocolName);
59  response->set("Sec-WebSocket-Accept", base64::serverWebSocketKey(request->header("sec-websocket-key")));
60 
61  response->status(101).end(); // Switch Protocol
62  } else {
63  response->set("Connection", "close");
64  response->status(400).end();
65  }
66  } else {
67  response->set("Sec-WebSocket-Version", "13");
68  response->set("Connection", "close");
69  response->status(400).end();
70  }
71 
72  return socketContext;
73  }
