58  SocketContextUpgradeFactory* socketContextUpgradeFactory = nullptr;
59 
60  std::string upgradeContextNames = req.header("upgrade");
61 
62  if (!upgradeContextNames.empty()) {
63  std::string upgradeContextName;
64  std::string upgradeContextPriority;
65

67  std::tie(upgradeContextName, upgradeContextPriority) = httputils::str_split(upgradeContextName, '/');
68 
69  httputils::to_lower(upgradeContextName);
70 
71  socketContextUpgradeFactory = select(upgradeContextName);
72 
73  if (socketContextUpgradeFactory != nullptr) {
74  socketContextUpgradeFactory->prepare(req, res);
