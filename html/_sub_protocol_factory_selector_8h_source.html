77  SubProtocolFactory* load(const std::string& subProtocolName, Role role) {
78  SubProtocolFactory* subProtocolFactory = nullptr;
79 
80  for (const std::string& searchPath : searchPaths) {
81  void* handle =
82  core::DynamicLoader::dlOpen(searchPath + "/libsnodec-websocket-" + subProtocolName + ".so", RTLD_LAZY | RTLD_LOCAL);
83 
84  if (handle != nullptr) {
85  SubProtocolFactory* (*getSubProtocolFactory)() = core::DynamicLoader::dlSym<SubProtocolFactory* (*) ()>(
86  handle, subProtocolName + (role == Role::SERVER ? "Server" : "Client") + "SubProtocolFactory");
87  if (getSubProtocolFactory != nullptr) {
