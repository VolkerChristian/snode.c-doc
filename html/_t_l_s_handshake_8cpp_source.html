59 
60  int sslErr = SSL_ERROR_NONE;

63  }
64 
65  switch (sslErr) {

67  ReadEventReceiver::resume();
68  break;
69  case SSL_ERROR_WANT_WRITE:
70  WriteEventReceiver::resume();
71  break;
72  case SSL_ERROR_NONE:
73  ReadEventReceiver::disable();
74  WriteEventReceiver::disable();
75  onSuccess();
76  break;
77  default:
78  ReadEventReceiver::disable();
