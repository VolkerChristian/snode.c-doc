151  template <typename SocketAddress>
152  ssize_t Socket<SocketAddress>::read_fd(void* ptr, size_t nbytes, int* recvfd) {
153  union {
154  struct cmsghdr cm;
155  char control[CMSG_SPACE(sizeof(int))] = {};
156  } control_un;
157 
158  msghdr msg;
159  msg.msg_control = control_un.control;
160  msg.msg_controllen = sizeof(control_un.control);
161 
162  msg.msg_name = nullptr;
163  msg.msg_namelen = 0;
164 
165  iovec iov[1];
166  msg.msg_iov = iov;
167  msg.msg_iovlen = 1;
168  msg.msg_iov[0].iov_base = ptr;
169  msg.msg_iov[0].iov_len = nbytes;
170 
171  ssize_t n = 0;
172 
173  if ((n = recvmsg(getFd(), &msg, 0)) > 0) {
174  cmsghdr* cmptr;
175 
176  if ((cmptr = CMSG_FIRSTHDR(&msg)) != nullptr && cmptr->cmsg_len == CMSG_LEN(sizeof(int))) {
177  if (cmptr->cmsg_level != SOL_SOCKET || cmptr->cmsg_type != SCM_RIGHTS) {
178  errno = EBADE;
179  *recvfd = -1;
180  n = -1;
181  } else {
182  *recvfd = *reinterpret_cast<int*>(CMSG_DATA(cmptr));
183  }
184  } else {
185  errno = ENOMSG;
186  *recvfd = -1;
187  n = -1;
188  }
189  }
190 
191  return n;
192  }
