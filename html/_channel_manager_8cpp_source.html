40  void ChannelManager::subscribe(const std::string& channel, SubProtocol* subProtocol) {
41  if (subProtocol->channel != channel) {
42  std::string newChannel = subProtocol->getName() + "/" + channel;
43 
44  channels[newChannel].insert(subProtocol);
45 
46  if (!subProtocol->channel.empty()) {
47  unsubscribe(subProtocol);
48  }
49 
50  subProtocol->channel = newChannel;
51  }
52  }
