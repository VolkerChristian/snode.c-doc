48  SocketAcceptor(const std::shared_ptr<SocketContextFactory>& socketContextFactory,
49  const std::function<void(const SocketAddress&, const SocketAddress&)>& onConnect,
50  const std::function<void(SocketConnection*)>& onConnected,
51  const std::function<void(SocketConnection*)>& onDisconnect,
52  const std::map<std::string, std::any>& options)
53  : core::socket::stream::SocketAcceptor<SocketConnection>(
54  socketContextFactory,
55  onConnect,
56  [onConnected, this](SocketConnection* socketConnection) -> void {
57  SSL* ssl = socketConnection->startSSL(this->masterSslCtx);
58 
59  if (ssl != nullptr) {
60  SSL_CTX_set_tlsext_servername_arg(this->masterSslCtx, this);
61 
62  SSL_set_accept_state(ssl);
63 
64  socketConnection->doSSLHandshake(
65  [&onConnected, socketConnection](void) -> void { // onSuccess
66  LOG(INFO) << "SSL/TLS initial handshake success";
67  socketConnection->SocketConnection::SocketReader::resume();
68  onConnected(socketConnection);
69  },
70  [](void) -> void { // onTimeout
71  LOG(WARNING) << "SSL/TLS initial handshake timed out";
72  },
73  [](int sslErr) -> void { // onError
74  ssl_log("SSL/TLS initial handshake failed", sslErr);
75  });
76  } else {
77  socketConnection->SocketConnection::SocketReader::disable();
78  socketConnection->SocketConnection::SocketWriter::disable();
79  ssl_log_error("SSL/TLS initialization failed");
80  }
81  },
82  [onDisconnect](SocketConnection* socketConnection) -> void { // onDisconnect
83  socketConnection->stopSSL();
84  onDisconnect(socketConnection);
85  },
86  options) {
87  masterSslCtx = ssl_ctx_new(options, true);
88 
89  if (masterSslCtx != nullptr) {
90  SSL_CTX_set_tlsext_servername_callback(masterSslCtx, serverNameCallback);
91  addMasterCert(masterSslCtx);
92  }
93 
94  sniSslCtxs = std::any_cast<std::shared_ptr<std::map<std::string, SSL_CTX*>>>(options.find("SNI_SSL_CTXS")->second);
95  }

101  void listen(const SocketAddress& localAddress, int backlog, const std::function<void(int)>& onError) {

111  private:
112  void addMasterCert(SSL_CTX* sSlCtx) {
113  if (sSlCtx != nullptr) {
114  X509* x509 = SSL_CTX_get0_certificate(sSlCtx);
115  if (x509 != nullptr) {
116  GENERAL_NAMES* subjectAltNames =
117  static_cast<GENERAL_NAMES*>(X509_get_ext_d2i(x509, NID_subject_alt_name, nullptr, nullptr));
118 
119  int32_t altNameCount = sk_GENERAL_NAME_num(subjectAltNames);
120 
121  for (int32_t i = 0; i < altNameCount; ++i) {
122  GENERAL_NAME* generalName = sk_GENERAL_NAME_value(subjectAltNames, i);
123  if (generalName->type == GEN_DNS) {
124  std::string subjectAltName =
125  std::string(reinterpret_cast<const char*>(ASN1_STRING_get0_data(generalName->d.dNSName)),
126  static_cast<std::size_t>(ASN1_STRING_length(generalName->d.dNSName)));
127  VLOG(2) << "SSL_CTX for domain '" << subjectAltName << "' installed";
128  masterSslCtxDomains.insert(subjectAltName);
129  }
130  }
131  sk_GENERAL_NAME_pop_free(subjectAltNames, GENERAL_NAME_free);
132  } else {
133  VLOG(2) << "\tClient certificate: no certificate";
134  }
135  }
136  }

138  static int serverNameCallback(SSL* ssl, [[maybe_unused]] int* al, [[maybe_unused]] void* arg) {
139  int ret = SSL_TLSEXT_ERR_OK;
140 
141  SocketAcceptor* socketAcceptor = static_cast<SocketAcceptor*>(arg);
142  std::shared_ptr<std::map<std::string, SSL_CTX*>> sniSslCtxs = socketAcceptor->sniSslCtxs;
143 
144  if (SSL_get_servername_type(ssl) != -1) {
145  std::string serverNameIndication = SSL_get_servername(ssl, TLSEXT_NAMETYPE_host_name);
146 
147  if (!socketAcceptor->masterSslCtxDomains.contains(serverNameIndication)) {
148  if (sniSslCtxs->contains(serverNameIndication)) {
149  SSL_CTX* sniSslCtx = (*sniSslCtxs.get())[serverNameIndication];
150 
151  SSL_CTX* nowUsedSslCtx = SSL_set_SSL_CTX(ssl, sniSslCtx);
152  if (nowUsedSslCtx == sniSslCtx) {
153  LOG(INFO) << "SSL_CTX: Switched for SNI '" << serverNameIndication << "'";
154  } else if (nowUsedSslCtx != nullptr) {
155  LOG(INFO) << "SSL_CTX: Not switcher for SNI '" << serverNameIndication << "': still using master SSL_CTX";
156  ret = SSL_TLSEXT_ERR_NOACK;
157  } else {
158  LOG(WARNING) << "SSL_CTX: Found but none used for SNI '" << serverNameIndication << '"';
159  }
160  } else {
161  LOG(INFO) << "SSL_CTX: Not found for SNI '" << serverNameIndication;
162  ret = SSL_TLSEXT_ERR_NOACK;
163  }
164  } else {
165  LOG(INFO) << "SSL_CTX: Master SSL_CTX already provides: '" << serverNameIndication << "'";
166  }
167  }
168 
169  return ret;
170  }
171 
172  private:
173  SSL_CTX* masterSslCtx = nullptr;
174  std::set<std::string> masterSslCtxDomains;
175 
176  std::shared_ptr<std::map<std::string, SSL_CTX*>> sniSslCtxs;
177  };
