35  template <typename SocketT>
36  class SocketAcceptor : public net::socket::stream::SocketAcceptor<net::socket::stream::tls::SocketConnection<SocketT>> {
37  public:
38  using SocketConnection = net::socket::stream::tls::SocketConnection<SocketT>;
39  using Socket = typename SocketConnection::Socket;
40  using SocketAddress = typename Socket::SocketAddress;

42  SocketAcceptor(const std::shared_ptr<SocketContextFactory>& socketContextFactory,
43  const std::function<void(const SocketAddress&, const SocketAddress&)>& onConnect,
44  const std::function<void(SocketConnection*)>& onConnected,
45  const std::function<void(SocketConnection*)>& onDisconnect,
46  const std::map<std::string, std::any>& options)
47  : net::socket::stream::SocketAcceptor<SocketConnection>(
48  socketContextFactory,
49  onConnect,
50  [onConnected, this](SocketConnection* socketConnection) -> void {
51  SSL* ssl = socketConnection->startSSL(this->ctx);
52 
53  if (ssl != nullptr) {
54  SSL_CTX_set_tlsext_servername_arg(this->ctx, socketConnection);
55 
56  SSL_set_accept_state(ssl);
57 
58  socketConnection->doSSLHandshake(
59  [&onConnected, socketConnection](void) -> void { // onSuccess
60  LOG(INFO) << "SSL/TLS initial handshake success";
61  socketConnection->SocketConnection::SocketReader::resume();
62  onConnected(socketConnection);
63  },
64  [](void) -> void { // onTimeout
65  LOG(WARNING) << "SSL/TLS initial handshake timed out";
66  },
67  [](int sslErr) -> void { // onError
68  ssl_log("SSL/TLS initial handshake failed", sslErr);
69  });
70  } else {
71  socketConnection->SocketConnection::SocketReader::disable();
72  socketConnection->SocketConnection::SocketWriter::disable();
73  ssl_log_error("SSL/TLS initialization failed");
74  }
75  },
76  [onDisconnect](SocketConnection* socketConnection) -> void { // onDisconnect
77  socketConnection->stopSSL();
78  onDisconnect(socketConnection);
79  },
80  options) {
81  ctx = ssl_ctx_new(options, true);
82  if (ctx != nullptr) {
83  SSL_CTX_set_tlsext_servername_callback(ctx, serverNameCallback);
84  }
85  }
86 
87  ~SocketAcceptor() override {
88  ssl_ctx_free(ctx);
89  }
90 
91  void listen(const SocketAddress& localAddress, int backlog, const std::function<void(int)>& onError) {
92  if (ctx == nullptr) {
93  errno = EINVAL;
94  onError(errno);
95  net::socket::stream::SocketAcceptor<SocketConnection>::destruct();
96  } else {
97  net::socket::stream::SocketAcceptor<SocketConnection>::listen(localAddress, backlog, onError);
98  }
99  }
100 
101  protected:
102  static int serverNameCallback(SSL* ssl, [[maybe_unused]] int* al, void* arg) {
103  SocketConnection* socketConnection = static_cast<SocketConnection*>(arg);
104 
105  if (SSL_get_servername_type(ssl) != -1) {
106  socketConnection->serverNameIndication = SSL_get_servername(ssl, TLSEXT_NAMETYPE_host_name);
107  LOG(INFO) << "ServerNameIndication: " << socketConnection->serverNameIndication;
108  }
109 
110  return SSL_TLSEXT_ERR_OK;
111  }
112 
113  SSL_CTX* ctx = nullptr;
114  };
