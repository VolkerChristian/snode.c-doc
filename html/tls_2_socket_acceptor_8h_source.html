52  SocketAcceptor(const std::shared_ptr<core::socket::SocketContextFactory>& socketContextFactory,
53  const std::function<void(SocketConnection*)>& onConnect,
54  const std::function<void(SocketConnection*)>& onConnected,
55  const std::function<void(SocketConnection*)>& onDisconnect,
56  const std::map<std::string, std::any>& options)
57  : Super(
58  socketContextFactory,
59  onConnect,
60  [onConnected, this](SocketConnection* socketConnection) -> void {
61  SSL* ssl = socketConnection->startSSL(
62  this->masterSslCtx, this->config->getInitTimeout(), this->config->getShutdownTimeout());
63 
64  if (ssl != nullptr) {
65  SSL_CTX_set_tlsext_servername_arg(this->masterSslCtx, this);
66 
67  SSL_set_accept_state(ssl);
68 
69  socketConnection->doSSLHandshake(
70  [&onConnected, socketConnection](void) -> void { // onSuccess
71  LOG(INFO) << "SSL/TLS initial handshake success";
72  onConnected(socketConnection);
73  },
74  [](void) -> void { // onTimeout
75  LOG(WARNING) << "SSL/TLS initial handshake timed out";
76  },
77  [](int sslErr) -> void { // onError
78  ssl_log("SSL/TLS initial handshake failed", sslErr);
79  });
80  } else {
81  socketConnection->close();
82  ssl_log_error("SSL/TLS initialization failed");
83  }
84  },
85  [onDisconnect](SocketConnection* socketConnection) -> void { // onDisconnect
86  socketConnection->stopSSL();
87  onDisconnect(socketConnection);
88  },
89  options) {
90  masterSslCtx = ssl_ctx_new(options, true);
91 
92  if (masterSslCtx != nullptr) {
93  SSL_CTX_set_tlsext_servername_callback(masterSslCtx, serverNameCallback);
94  addMasterCtx(masterSslCtx);
95  }

98  forceSni = *std::any_cast<bool*>(options.find("FORCE_SNI")->second);
99  }

105  void listen(const std::shared_ptr<Config>& config, const std::function<void(const SocketAddress&, int)>& onError) {

115  private:
116  void addMasterCtx(SSL_CTX* sslCtx) {
117  if (sslCtx != nullptr) {
118  X509* x509 = SSL_CTX_get0_certificate(sslCtx);
119  if (x509 != nullptr) {
120  GENERAL_NAMES* subjectAltNames =
121  static_cast<GENERAL_NAMES*>(X509_get_ext_d2i(x509, NID_subject_alt_name, nullptr, nullptr));
122 
123  int32_t altNameCount = sk_GENERAL_NAME_num(subjectAltNames);
124 
125  for (int32_t i = 0; i < altNameCount; ++i) {
126  GENERAL_NAME* generalName = sk_GENERAL_NAME_value(subjectAltNames, i);
127  if (generalName->type == GEN_DNS) {
128  std::string subjectAltName =
129  std::string(reinterpret_cast<const char*>(ASN1_STRING_get0_data(generalName->d.dNSName)),
130  static_cast<std::size_t>(ASN1_STRING_length(generalName->d.dNSName)));
131  VLOG(2) << "SSL_CTX for domain '" << subjectAltName << "' installed";
132  masterSslCtxDomains.insert(subjectAltName);
133  }
134  }
135  sk_GENERAL_NAME_pop_free(subjectAltNames, GENERAL_NAME_free);
136  } else {
137  VLOG(2) << "\tClient certificate: no certificate";
138  }
139  }
140  }

142  static int serverNameCallback(SSL* ssl, [[maybe_unused]] int* al, [[maybe_unused]] void* arg) {
143  int ret = SSL_TLSEXT_ERR_OK;
144 
145  SocketAcceptor* socketAcceptor = static_cast<SocketAcceptor*>(arg);
146 
147  if (SSL_get_servername_type(ssl) != -1) {
148  std::string serverNameIndication = SSL_get_servername(ssl, TLSEXT_NAMETYPE_host_name);
149 
150  if (!socketAcceptor->masterSslCtxDomains.contains(serverNameIndication)) {
151  if (socketAcceptor->sniSslCtxs->contains(serverNameIndication)) {
152  SSL_CTX* sniSslCtx = (*socketAcceptor->sniSslCtxs.get())[serverNameIndication];
153 
154  SSL_CTX* nowUsedSslCtx = SSL_set_SSL_CTX(ssl, sniSslCtx);
155 
156  if (nowUsedSslCtx == sniSslCtx) {
157  LOG(INFO) << "SSL_CTX: Switched for SNI '" << serverNameIndication << "'";
158  } else if (nowUsedSslCtx != nullptr) {
159  if (!socketAcceptor->forceSni) {
160  LOG(WARNING) << "SSL_CTX: Not switcher for SNI '" << serverNameIndication
161  << "'. Master SSL_CTX still used.";
162  ret = SSL_TLSEXT_ERR_ALERT_WARNING;
163  } else {
164  LOG(ERROR) << "SSL_CTX: Not switcher for SNI '" << serverNameIndication << "'.";
165  ret = SSL_TLSEXT_ERR_ALERT_FATAL;
166  }
167  } else {
168  if (!socketAcceptor->forceSni) {
169  LOG(WARNING) << "SSL_CTX: Not switcher for SNI '" << serverNameIndication
170  << "'. Master SSL_CTX still used.";
171  ret = SSL_TLSEXT_ERR_ALERT_WARNING;
172  } else {
173  LOG(ERROR) << "SSL_CTX: Found but none used for SNI '" << serverNameIndication << '"';
174  ret = SSL_TLSEXT_ERR_ALERT_FATAL;
175  }
176  }
177  } else {
178  if (!socketAcceptor->forceSni) {
179  LOG(WARNING) << "SSL_CTX: Not found for SNI '" << serverNameIndication << "'. Master SSL_CTX still used.";
180  } else {
181  LOG(ERROR) << "SSL_CTX: Not found for SNI '" << serverNameIndication << "'.";
182  ret = SSL_TLSEXT_ERR_ALERT_FATAL;
183  }
184  }
185  } else {
186  LOG(INFO) << "SSL_CTX: Master SSL_CTX already provides SNI '" << serverNameIndication << "'";
187  }
188  }
189 
190  return ret;
191  }

193  private:

196 
197  std::shared_ptr<std::map<std::string, SSL_CTX*>> sniSslCtxs;
198  bool forceSni = false;
199  };
