107  void sendClose(uint16_t statusCode = 1000, const char* reason = nullptr, std::size_t reasonLength = 0) {
108  char* closePayload = const_cast<char*>(reason);
109  std::size_t closePayloadLength = reasonLength;
110 
111  if (statusCode != 0) {
112  closePayload = new char[reasonLength + 2];
113  *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
114  closePayloadLength += 2;
115  if (reasonLength > 0) {
116  memcpy(closePayload + 2, reason, reasonLength);
117  }
118  }
119 
120  sendClose(closePayload, closePayloadLength);
121 
122  if (statusCode != 0) {
123  delete[] closePayload;
124  }
125 
126  setTimeout(CLOSE_SOCKET_TIMEOUT);
127 
128  closeSent = true;
129  }

154  void onMessageData(const char* junk, uint64_t junkLen) override {
155  switch (opCodeReceived) {
156  case OpCode::CLOSE:
157  [[fallthrough]];
158  case OpCode::PING:
159  [[fallthrough]];
160  case OpCode::PONG:
161  pongCloseData += std::string(junk, static_cast<std::size_t>(junkLen));
162  break;
163  default:
164  std::size_t junkOffset = 0;
165 
166  do {
167  std::size_t sendJunkLen =
168  (junkLen - junkOffset <= SIZE_MAX) ? static_cast<std::size_t>(junkLen - junkOffset) : SIZE_MAX;
169  subProtocol->onMessageData(junk + junkOffset, sendJunkLen);
170  junkOffset += sendJunkLen;
171  } while (junkLen - junkOffset > 0);
172  break;
173  }
174  }
175 
176  void onMessageEnd() override {
177  switch (opCodeReceived) {
178  case OpCode::CLOSE:
179  if (closeSent) { // active close
180  closeSent = false;
181  VLOG(0) << "Close confirmed from peer";
182  } else { // passive close
183  VLOG(0) << "Close request received - replying with close";
184  sendClose(pongCloseData.data(), pongCloseData.length());
185  pongCloseData.clear();
186  }
187  break;
188  case OpCode::PING:
189  sendPong(pongCloseData.data(), pongCloseData.length());
190  pongCloseData.clear();
191  break;
192  case OpCode::PONG:
193  subProtocol->onPongReceived();
194  break;
195  default:
196  subProtocol->onMessageEnd();
197  break;
198  }
199  }
