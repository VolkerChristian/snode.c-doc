36  bool MiddlewareDispatcher::dispatch(express::State& state, const std::string& parentMountPath, const MountPoint& mountPoint) {
37  bool dispatched = false;
38 
39  if ((state.getFlags() & State::NEXT) == 0) {
40  std::string absoluteMountPath = path_concat(parentMountPath, mountPoint.relativeMountPath);
41 
42  if ((state.getRequest()->path.rfind(absoluteMountPath, 0) == 0 && mountPoint.method == "use") ||
43  ((absoluteMountPath == state.getRequest()->path || checkForUrlMatch(absoluteMountPath, state.getRequest()->url)) &&
44  (state.getRequest()->method == mountPoint.method || mountPoint.method == "all"))) {
45  dispatched = true;
46 
47  if (hasResult(absoluteMountPath)) {
48  setParams(absoluteMountPath, *state.getRequest());
49  }
50 
51  Next next(state);
52  lambda(*state.getRequest(), *state.getResponse(), next);
53 
54  // If next() was called synchroneously continue current route-tree traversal
55  if ((next.state.getFlags() & express::State::NEXT) != 0) {
56  dispatched = false;
57  state = next.state;
58  }
59  }
60  }
61 
62  return dispatched;
63  }
