87  void initConnectEvent() override {
88  socket = new PrimarySocket();
89  if (socket->open(PrimarySocket::SOCK::NONBLOCK) < 0) {
90  onError(config->getRemoteAddress(), errno);
91  destruct();
92  } else if (socket->bind(config->getLocalAddress()) < 0) {
93  onError(config->getRemoteAddress(), errno);
94  destruct();
95  } else if (socket->connect(config->getRemoteAddress()) < 0 && !socket->connectInProgress(errno)) {
96  onError(config->getRemoteAddress(), errno);
97  destruct();
98  } else {
99  enable(socket->getFd());
100  }

103  void connectEvent() override {
104  int cErrno = -1;
105 
106  if ((cErrno = socket->getSockError()) >= 0) { // >= 0->return valid : < 0->getsockopt failed errno = cErrno;
107  if (!socket->connectInProgress(cErrno)) {
108  if (cErrno == 0) {
109  disable();
110  socketConnectionFactory.create(*socket, config);
111  errno = errno == 0 ? cErrno : errno;
112  onError(config->getRemoteAddress(), cErrno);
113  } else {
114  disable();
115  errno = errno == 0 ? cErrno : errno;
116  onError(config->getRemoteAddress(), cErrno);
117  }
118  } else {
119  // Do nothing: connect() still in progress
120  }
121  } else {
122  disable();
123  onError(config->getRemoteAddress(), cErrno);
124  }
125  }
