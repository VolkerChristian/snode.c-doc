87  void initConnectEvent() override {
88  Socket::open(
89  [this](int errnum) -> void {
90  if (errnum > 0) {
91  onError(config->getRemoteAddress(), errnum);
92  destruct();
93  } else {
94  Socket::bind(config->getLocalAddress(), [this](int errnum) -> void {
95  if (errnum > 0) {
96  onError(config->getRemoteAddress(), errnum);
97  destruct();
98  } else {
99  int ret = core::system::connect(Socket::getFd(),
100  &config->getRemoteAddress().getSockAddr(),
101  config->getRemoteAddress().getSockAddrLen());
102 
103  if (ret == 0 || errno == EINPROGRESS) {
104  enable(Socket::getFd());
105  onError(config->getRemoteAddress(), 0);
106  } else {
107  onError(config->getRemoteAddress(), errno);
108  destruct();
109  }
110  }
111  });
112  }
113  },

117  void connectEvent() override {
118  int cErrno = -1;
119  socklen_t cErrnoLen = sizeof(cErrno);
120 
121  int err = core::system::getsockopt(SocketConnector::getFd(), SOL_SOCKET, SO_ERROR, &cErrno, &cErrnoLen);
122 
123  if (err == 0) {
124  errno = cErrno;
125  if (errno != EINPROGRESS) {
126  if (errno == 0) {
127  disable();
128 
129  typename SocketAddress::SockAddr localAddress{};
130  socklen_t localAddressLength = sizeof(localAddress);
131 
132  typename SocketAddress::SockAddr remoteAddress{};
133  socklen_t remoteAddressLength = sizeof(remoteAddress);
134 
135  if (core::system::getsockname(Socket::getFd(), reinterpret_cast<sockaddr*>(&localAddress), &localAddressLength) ==
136  0 &&
137  core::system::getpeername(Socket::getFd(), reinterpret_cast<sockaddr*>(&remoteAddress), &remoteAddressLength) ==
138  0) {
139  SocketConnection* socketConnection = new SocketConnection(SocketConnector::getFd(),
140  socketContextFactory,
141  SocketAddress(localAddress),
142  SocketAddress(remoteAddress),
143  onConnect,
144  onDisconnect,
145  config->getReadTimeout(),
146  config->getWriteTimeout(),
147  config->getReadBlockSize(),
148  config->getWriteBlockSize(),
149  config->getTerminateTimeout());
150 
151  onConnected(socketConnection);
152 
153  Socket::dontClose(true);
154  } else {
155  onError(config->getRemoteAddress(), errno);
156  disable();
157  }
158  } else {
159  onError(config->getRemoteAddress(), errno);
160  disable();
161  }
162  } else {
163  // Do nothing: connect() still in progress
164  }
165  } else {
166  onError(config->getRemoteAddress(), errno);

168  }
169  }
