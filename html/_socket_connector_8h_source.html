69  void connect(const SocketAddress& remoteAddress, const SocketAddress& bindAddress, const std::function<void(int)>& onError) {
70  this->onError = onError;
71 
72  Socket::open(
73  [this, &bindAddress, &remoteAddress, &onError](int errnum) -> void {
74  if (errnum > 0) {
75  onError(errnum);
76  destruct();
77  } else {
78  Socket::bind(bindAddress, [this, &remoteAddress, &onError](int errnum) -> void {
79  if (errnum > 0) {
80  onError(errnum);
81  destruct();
82  } else {
83  int ret = core::system::connect(Socket::fd, &remoteAddress.getSockAddr(), remoteAddress.getSockAddrLen());
84 
85  if (ret == 0 || errno == EINPROGRESS) {
86  enable(Socket::fd);
87  } else {
88  onError(errno);
89  destruct();
90  }
91  }
92  });
93  }
94  },
95  SOCK_NONBLOCK);
96  }

99  void connectEvent() override {
100  int cErrno = -1;
101  socklen_t cErrnoLen = sizeof(cErrno);
102 
103  int err = core::system::getsockopt(SocketConnector::getFd(), SOL_SOCKET, SO_ERROR, &cErrno, &cErrnoLen);
104 
105  if (err == 0) {
106  errno = cErrno;
107  if (errno != EINPROGRESS) {
108  if (errno == 0) {
109  typename SocketAddress::SockAddr localAddress{};
110  socklen_t localAddressLength = sizeof(localAddress);
111 
112  typename SocketAddress::SockAddr remoteAddress{};
113  socklen_t remoteAddressLength = sizeof(remoteAddress);
114 
115  if (core::system::getsockname(Socket::fd, reinterpret_cast<sockaddr*>(&localAddress), &localAddressLength) == 0 &&
116  core::system::getpeername(Socket::fd, reinterpret_cast<sockaddr*>(&remoteAddress), &remoteAddressLength) == 0) {
117  SocketConnection* socketConnection = new SocketConnection(SocketConnector::getFd(),
118  socketContextFactory,
