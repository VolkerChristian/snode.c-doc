76  void connect(const SocketAddress& remoteAddress, const SocketAddress& bindAddress, const std::function<void(int)>& onError) {
77  this->onError = onError;
78 
79  Socket::open(
80  [this, &bindAddress, &remoteAddress, &onError](int errnum) -> void {
81  if (errnum > 0) {
82  onError(errnum);
83  destruct();
84  } else {
85  Socket::bind(bindAddress, [this, &remoteAddress, &onError](int errnum) -> void {
86  if (errnum > 0) {
87  onError(errnum);
88  destruct();
89  } else {
90  int ret = core::system::connect(Socket::fd, &remoteAddress.getSockAddr(), remoteAddress.getSockAddrLen());
91 
92  if (ret == 0 || errno == EINPROGRESS) {
93  enable(Socket::fd);
94  } else {
95  onError(errno);
96  destruct();
97  }
98  }
99  });
100  }
101  },
102  SOCK_NONBLOCK);
103  }

110  int err = core::system::getsockopt(SocketConnector::getFd(), SOL_SOCKET, SO_ERROR, &cErrno, &cErrnoLen);
111 
112  if (err == 0) {
113  errno = cErrno;
114  if (errno != EINPROGRESS) {
115  if (errno == 0) {
116  typename SocketAddress::SockAddr localAddress{};
117  socklen_t localAddressLength = sizeof(localAddress);
118 
119  typename SocketAddress::SockAddr remoteAddress{};
120  socklen_t remoteAddressLength = sizeof(remoteAddress);
121 
122  if (core::system::getsockname(Socket::fd, reinterpret_cast<sockaddr*>(&localAddress), &localAddressLength) == 0 &&
