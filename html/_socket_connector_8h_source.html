86  void initConnectEvent() override {
87  Socket::open(
88  [this](int errnum) -> void {
89  if (errnum > 0) {
90  onError(config->getRemoteAddress(), errnum);
91  destruct();
92  } else {
93  Socket::bind(config->getLocalAddress(), [this](int errnum) -> void {
94  if (errnum > 0) {
95  onError(config->getRemoteAddress(), errnum);
96  destruct();
97  } else {
98  int ret = core::system::connect(Socket::getFd(),
99  &config->getRemoteAddress().getSockAddr(),
100  config->getRemoteAddress().getSockAddrLen());
101 
102  if (ret == 0 || errno == EINPROGRESS) {
103  enable(Socket::getFd());
104  onError(config->getRemoteAddress(), 0);
105  } else {
106  onError(config->getRemoteAddress(), errno);
107  destruct();
108  }
109  }
110  });
111  }
112  },
113  SOCK_NONBLOCK);

116  void connectEvent() override {
117  int cErrno = -1;
118  socklen_t cErrnoLen = sizeof(cErrno);
119 
120  int err = core::system::getsockopt(SocketConnector::getFd(), SOL_SOCKET, SO_ERROR, &cErrno, &cErrnoLen);
121 
122  if (err == 0) {
123  errno = cErrno;
124  if (errno != EINPROGRESS) {
125  if (errno == 0) {
126  typename SocketAddress::SockAddr localAddress{};
127  socklen_t localAddressLength = sizeof(localAddress);
128 
129  typename SocketAddress::SockAddr remoteAddress{};
130  socklen_t remoteAddressLength = sizeof(remoteAddress);
131 
132  if (core::system::getsockname(Socket::getFd(), reinterpret_cast<sockaddr*>(&localAddress), &localAddressLength) ==
133  0 &&
134  core::system::getpeername(Socket::getFd(), reinterpret_cast<sockaddr*>(&remoteAddress), &remoteAddressLength) ==
135  0) {
136  SocketConnection* socketConnection = new SocketConnection(SocketConnector::getFd(),
137  socketContextFactory,
138  SocketAddress(localAddress),
139  SocketAddress(remoteAddress),
140  onConnect,
141  onDisconnect,
142  config->getReadTimeout(),
143  config->getWriteTimeout(),
144  config->getReadBlockSize(),
145  config->getWriteBlockSize(),
146  config->getTerminateTimeout());
147 
148  onConnected(socketConnection);
149  onError(config->getRemoteAddress(), 0);
150 
151  Socket::dontClose(true);
152  disable();
153  } else {
154  onError(config->getRemoteAddress(), errno);
155  disable();
156  }
157  } else {
158  onError(config->getRemoteAddress(), errno);
159  disable();
160  }
161  } else {
162  // Do nothing: connect() still in progress
163  }
164  } else {
165  onError(config->getRemoteAddress(), errno);
166  disable();
167  }
168  }
