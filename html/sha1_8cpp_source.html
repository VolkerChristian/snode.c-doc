34 /* (R0+R1), R2, R3, R4 are the different operations used in SHA1 */
35 #define SHA1_R0(v, w, x, y, z, i) \
36  z += ((w & (x ^ y)) ^ y) + block[i] + 0x5a827999 + SHA1_ROL(v, 5); \
37  w = SHA1_ROL(w, 30);
38 #define SHA1_R1(v, w, x, y, z, i) \

77 std::string SHA1::final() {
78  /* Total number of hashed bits */
79  uint64_t total_bits = (transforms * BLOCK_BYTES + buffer.size()) * 8;
80 
81  /* Padding */
82  buffer += static_cast<std::string::value_type>(0x80);
83  std::size_t orig_size = buffer.size();
84  while (buffer.size() < BLOCK_BYTES) {
85  buffer += static_cast<char>(0x00);
86  }
87 
88  uint32_t block[BLOCK_INTS];
89  buffer_to_block(buffer, block);
90 
91  if (orig_size > BLOCK_BYTES - 8) {
92  transform(block);
93  for (unsigned int i = 0; i < BLOCK_INTS - 2; i++) {
94  block[i] = 0;
95  }
96  }
97 
98  /* Append total_bits, split this uint64 into two uint32 */
99  block[BLOCK_INTS - 1] = static_cast<uint32_t>(total_bits & 0x00000000FFFFFFFF);
100  block[BLOCK_INTS - 2] = static_cast<uint32_t>((total_bits >> 32) & 0x00000000FFFFFFFF);
101  transform(block);
102 
103  /* Hex std::string */
104  std::ostringstream result;
105  for (unsigned int i = 0; i < DIGEST_INTS; i++) {
106  result << std::hex << std::setfill('0') << std::setw(8);
107  result << (digest[i] & 0xffffffff);
108  }
109 
110  /* Reset for next run */
111  reset();
112 
113  return result.str();
114 }
