89  int DynamicLoader::execDlClose(void* handle) {
90  int ret = -1;
91 
92  if (handle != nullptr) {
93  if (dlOpenedLibraries.contains(handle)) {
94  VLOG(0) << "execDLClose: " << dlOpenedLibraries[handle].fileName;
95 
96  ret = core::system::dlclose(handle);
97  dlOpenedLibraries[handle].refCount--;
98 
99  if (dlOpenedLibraries[handle].refCount == 0) {
100  dlOpenedLibraries.erase(handle);
101  }
102  }
103  }
104 
105  return ret;

108  void DynamicLoader::execDlCloseDeleyed() {
109  for (auto& [handle, refCount] : registeredForDlClose) {
