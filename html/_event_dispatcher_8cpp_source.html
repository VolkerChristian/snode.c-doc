81  TickStatus EventDispatcher::dispatch(const utils::Timeval& tickTimeOut, bool stopped) {
82  TickStatus tickStatus = TickStatus::SUCCESS;
83 
84  utils::Timeval currentTime = utils::Timeval::currentTime();
85 
86  executeEventQueue(currentTime);
87 
88  checkTimedOutEvents(currentTime);
89  unobserveDisabledEvents(currentTime);
90  observeEnabledEvents(currentTime);
91 
92  if (getObservedEventReceiverCount() > 0 || (!timerEventDispatcher->empty() && !stopped)) {
93  utils::Timeval nextEventTimeout = getNextTimeout(currentTime);
94  utils::Timeval nextTimerTimeout = timerEventDispatcher->getNextTimeout(currentTime);
95 
96  utils::Timeval nextTimeout = stopped ? nextEventTimeout : std::min(nextTimerTimeout, nextEventTimeout);
97 
98  nextTimeout = std::min(nextTimeout, tickTimeOut);
99  nextTimeout = std::max(nextTimeout, utils::Timeval()); // In case nextEventTimeout is negativ
100 
101  int ret = multiplex(nextTimeout);
102 
103  if (ret >= 0) {
104  currentTime = utils::Timeval::currentTime();
105 
106  timerEventDispatcher->dispatchActiveEvents(currentTime);
107 
108  dispatchActiveEvents(ret);
109  dispatchImmediateEvents();
110  } else {
111  if (errno != EINTR) {
112  tickStatus = TickStatus::ERROR;
113  }
114  }
115  } else {
116  tickStatus = TickStatus::NO_OBSERVER;
117  }
118 
119  return tickStatus;
120  }

122  void EventDispatcher::stop() {
123  core::TickStatus tickStatus;
124 
125  do {
126  for (core::DescriptorEventDispatcher* const eventDispatcher : descriptorEventDispatcher) {
127  eventDispatcher->stop();
128  }
129  tickStatus = dispatch(2, true);
130  } while (tickStatus == TickStatus::SUCCESS);
131 
132  do {
133  timerEventDispatcher->stop();
134  tickStatus = dispatch(0, false);
135  } while (tickStatus == TickStatus::SUCCESS);
