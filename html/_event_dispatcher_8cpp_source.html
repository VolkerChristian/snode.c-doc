68  void EventDispatcher::enable(EventReceiver* eventReceiver, int fd) {
69  if (disabledEventReceiver[fd].contains(eventReceiver)) {
70  // same tick as disable
71  disabledEventReceiver[fd].remove(eventReceiver);
72  } else if (!eventReceiver->isEnabled() &&
73  (!enabledEventReceiver.contains(fd) || !enabledEventReceiver[fd].contains(eventReceiver))) {
74  // next tick as disable
75  enabledEventReceiver[fd].push_back(eventReceiver);
76  eventReceiver->enabled();
77  if (unobservedEventReceiver.contains(eventReceiver)) {
78  unobservedEventReceiver.remove(eventReceiver);
79  }
80  } else {
81  LOG(WARNING) << "EventReceiver double enable";
82  }
83  }

142  struct timeval EventDispatcher::observeEnabledEvents() {
143  struct timeval nextTimeout = {LONG_MAX, 0};
144 
145  for (const auto& [fd, eventReceivers] : enabledEventReceiver) {
146  for (EventReceiver* eventReceiver : eventReceivers) {
147  observedEventReceiver[fd].push_front(eventReceiver);

161  struct timeval EventDispatcher::dispatchActiveEvents(struct timeval currentTime) {
162  struct timeval nextInactivityTimeout {
163  LONG_MAX, 0
164  };
165 
166  for (const auto& [fd, eventReceivers] : observedEventReceiver) {
167  EventReceiver* eventReceiver = eventReceivers.front();
168  struct timeval maxInactivity = eventReceiver->getTimeout();
169  if (fdSet.isSet(fd) || (eventReceiver->continueImmediately())) {
170  eventCounter++;
171  eventReceiver->dispatchEvent();
172  eventReceiver->triggered(currentTime);
173  if (!eventReceiver->isSuspended()) {
174  if (eventReceiver->continueImmediately()) {
175  nextInactivityTimeout = {0, 0};
176  } else {
177  nextInactivityTimeout = std::min(nextInactivityTimeout, maxInactivity);
178  }
179  }
180  } else {
181  struct timeval inactivity = currentTime - eventReceiver->getLastTriggered();
182  if (inactivity >= maxInactivity) {
183  eventReceiver->timeoutEvent();
184  } else {
185  nextInactivityTimeout = std::min(maxInactivity - inactivity, nextInactivityTimeout);
186  }
187  }
188  }
189 
190  return nextInactivityTimeout;
191  }

193  void EventDispatcher::unobserveDisabledEvents() {
194  for (const auto& [fd, eventReceivers] : disabledEventReceiver) {
195  for (EventReceiver* eventReceiver : eventReceivers) {
196  observedEventReceiver[fd].remove(eventReceiver);
197  if (observedEventReceiver[fd].empty() || observedEventReceiver[fd].front()->isSuspended()) {
198  if (observedEventReceiver[fd].empty()) {
199  observedEventReceiver.erase(fd);
200  }
201  fdSet.clr(fd);
202  } else {
203  fdSet.set(fd);
204  observedEventReceiver[fd].front()->triggered();
205  }
206  eventReceiver->disabled();
207  if (eventReceiver->observationCounter == 0) {
208  unobservedEventReceiver.push_back(eventReceiver);
209  }
210  eventReceiver->fd = -1;
211  }
212  }
213  disabledEventReceiver.clear();
214  }
