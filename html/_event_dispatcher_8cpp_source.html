76  TickStatus EventDispatcher::dispatch(const utils::Timeval& tickTimeOut, bool stopped) {
77  TickStatus tickStatus = TickStatus::SUCCESS;
78 
79  EventDispatcher::observeEnabledEvents();
80 
81  int observedEventReceiverCount = getObservedEventReceiverCount();
82 
83  utils::Timeval currentTime = utils::Timeval::currentTime();
84 
85  utils::Timeval nextEventTimeout = getNextTimeout(currentTime);
86  utils::Timeval nextTimerTimeout = timerEventDispatcher->getNextTimeout(currentTime);
87 
88  if (observedEventReceiverCount > 0 || (!timerEventDispatcher->empty() && !stopped)) {
89  utils::Timeval nextTimeout = stopped ? nextEventTimeout : std::min(nextTimerTimeout, nextEventTimeout);
90 
91  nextTimeout = std::min(nextTimeout, tickTimeOut);
92  nextTimeout = std::max(nextTimeout, utils::Timeval()); // In case nextEventTimeout is negativ
93 
94  int ret = multiplex(nextTimeout);
95 
96  if (ret >= 0) {
97  currentTime = utils::Timeval::currentTime();
98 
99  timerEventDispatcher->dispatchActiveEvents(currentTime);
100 
101  dispatchActiveEvents(ret, currentTime);
102  unobserveDisabledEvents(currentTime);
103  } else {
104  if (errno != EINTR) {
105  tickStatus = TickStatus::ERROR;
106  }
107  }
108  } else {
109  tickStatus = TickStatus::NO_OBSERVER;
110  }
111 
112  return tickStatus;
113  }
