71  void EventDispatcher::enable(EventReceiver* eventReceiver, int fd) {
72  if (disabledEventReceiver[fd].contains(eventReceiver)) {
73  // same tick as disable
74  disabledEventReceiver[fd].remove(eventReceiver);
75  } else if (!eventReceiver->isEnabled() &&
76  (!enabledEventReceiver.contains(fd) || !enabledEventReceiver[fd].contains(eventReceiver))) {
77  // next tick as disable
78  enabledEventReceiver[fd].push_back(eventReceiver);
79  eventReceiver->enabled();
80  if (unobservedEventReceiver.contains(eventReceiver)) {
81  unobservedEventReceiver.remove(eventReceiver);
82  }
83  } else {
84  LOG(WARNING) << "EventReceiver double enable";
85  }
86  }

145  struct timeval EventDispatcher::observeEnabledEvents() {
146  struct timeval nextTimeout = {LONG_MAX, 0};
147 
148  for (const auto& [fd, eventReceivers] : enabledEventReceiver) {
149  for (EventReceiver* eventReceiver : eventReceivers) {
150  observedEventReceiver[fd].push_front(eventReceiver);
151  if (!eventReceiver->isSuspended()) {
152  fdSet.set(fd);
153  nextTimeout = std::min(nextTimeout, eventReceiver->getTimeout());
154  } else {
155  fdSet.clr(fd);
156  }
157  }
158  }
159  enabledEventReceiver.clear();
160 
161  return nextTimeout;

164  struct timeval EventDispatcher::dispatchActiveEvents(struct timeval currentTime) {
165  struct timeval nextInactivityTimeout {
166  LONG_MAX, 0
167  };
168 
169  for (const auto& [fd, eventReceivers] : observedEventReceiver) {
170  EventReceiver* eventReceiver = eventReceivers.front();
171  struct timeval maxInactivity = eventReceiver->getTimeout();
172  if (fdSet.isSet(fd) || (eventReceiver->continueImmediately())) {
173  eventCounter++;
174  eventReceiver->dispatchEvent();
175  eventReceiver->triggered(currentTime);
176  if (!eventReceiver->isSuspended()) {
177  if (eventReceiver->continueImmediately()) {
178  nextInactivityTimeout = {0, 0};
179  } else {
180  nextInactivityTimeout = std::min(nextInactivityTimeout, maxInactivity);
181  }
182  }
183  } else {
184  struct timeval inactivity = currentTime - eventReceiver->getLastTriggered();
185  if (inactivity >= maxInactivity) {
186  eventReceiver->timeoutEvent();
187  } else {
188  nextInactivityTimeout = std::min(maxInactivity - inactivity, nextInactivityTimeout);
189  }
190  }
191  }
192 
193  return nextInactivityTimeout;
194  }

196  void EventDispatcher::unobserveDisabledEvents() {
197  for (const auto& [fd, eventReceivers] : disabledEventReceiver) {
198  for (EventReceiver* eventReceiver : eventReceivers) {
199  observedEventReceiver[fd].remove(eventReceiver);
200  if (observedEventReceiver[fd].empty() || observedEventReceiver[fd].front()->isSuspended()) {
201  if (observedEventReceiver[fd].empty()) {
202  observedEventReceiver.erase(fd);
203  }
204  fdSet.clr(fd);
205  } else {
206  fdSet.set(fd);
207  observedEventReceiver[fd].front()->triggered();
208  }
209  eventReceiver->disabled();
210  if (eventReceiver->observationCounter == 0) {
211  unobservedEventReceiver.push_back(eventReceiver);
212  }
213  eventReceiver->fd = -1;
214  }
215  }
216  disabledEventReceiver.clear();
217  }
