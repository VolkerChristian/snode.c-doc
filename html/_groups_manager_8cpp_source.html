39  void GroupsManager::subscribe(SubProtocol* subProtocol, std::string channel) {
40  if (channel.empty()) {
41  channel = subProtocol->getName();
42  }
43 
44  if (subProtocol->group != channel) {
45  std::string newChannel = subProtocol->getName() + "/" + channel;
46 
47  groups[newChannel].insert(subProtocol);
48 
49  if (!subProtocol->group.empty()) {
50  unsubscribe(subProtocol);
51  }
52 
53  subProtocol->group = newChannel;
54  }
55  }

57  void GroupsManager::unsubscribe(SubProtocol* subProtocol) {
58  if (groups.contains(subProtocol->group)) {
59  groups[subProtocol->group].erase(subProtocol);
60 
61  if (groups[subProtocol->group].size() == 0) {
62  groups.erase(subProtocol->group);
63  if (groups.size() == 0) {
64  delete this;
