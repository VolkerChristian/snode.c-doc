58  SocketAddress::SocketAddress(const std::string& ipOrHostname, uint16_t port) {
59  struct addrinfo hints {};
60  struct addrinfo* res;
61  struct addrinfo* resalloc;
62 
63  memset(&hints, 0, sizeof(hints));
64  memset(&sockAddr, 0, sizeof(sockAddr));
65 
66  /* We only care about IPV6 results */
67  hints.ai_family = AF_INET;
68  hints.ai_socktype = 0;
69  hints.ai_flags = AI_ADDRCONFIG;
70 
71  int err = core::system::getaddrinfo(ipOrHostname.c_str(), nullptr, &hints, &res);
72 
73  if (err != 0) {
74  throw bad_hostname(ipOrHostname);
75  }
76 
77  resalloc = res;
78 
79  while (res) {
80  /* Check to make sure we have a valid AF_INET6 address */
81  if (res->ai_family == AF_INET) {
82  /* Use memcpy since we're going to free the res variable later */
83  memcpy(&sockAddr, res->ai_addr, res->ai_addrlen);
84 
85  /* Here we convert the port to network byte order */
86  sockAddr.sin_port = htons(port);
87  sockAddr.sin_family = AF_INET;
88  break;
89  }
90 
91  res = res->ai_next;
92  }
93 
94  core::system::freeaddrinfo(resalloc);
95  }
