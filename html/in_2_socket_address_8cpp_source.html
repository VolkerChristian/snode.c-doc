60  void SocketAddress::setHost(const std::string& ipOrHostname) {
61  struct addrinfo hints {};
62  struct addrinfo* res;
63  struct addrinfo* resalloc;
64 
65  memset(&hints, 0, sizeof(hints));
66 
67  /* We only care about IPV results */
68  hints.ai_family = AF_INET;
69  hints.ai_socktype = 0;
70  hints.ai_flags = AI_ADDRCONFIG;
71 
72  int err = core::system::getaddrinfo(ipOrHostname.c_str(), nullptr, &hints, &res);
73 
74  if (err != 0) {
75  throw bad_hostname(ipOrHostname + ": " + gai_strerror(err));
76  }
77 
78  resalloc = res;
79 
80  while (res) {
81  /* Check to make sure we have a valid AF_INET address */
82  if (res->ai_family == AF_INET) {
83  sockAddr.sin_addr.s_addr = reinterpret_cast<sockaddr_in*>(res->ai_addr)->sin_addr.s_addr;
84  break;
85  }
86 
87  res = res->ai_next;
88  }
89 
90  core::system::freeaddrinfo(resalloc);
91  }
