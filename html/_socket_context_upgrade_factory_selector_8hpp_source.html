69  template <typename SocketContextUpgradeFactory>
70  SocketContextUpgradeFactory*
71  SocketContextUpgradeFactorySelector<SocketContextUpgradeFactory>::load(const std::string& upgradeContextName,
72  typename SocketContextUpgradeFactory::Role role) {
73  SocketContextUpgradeFactory* socketContextUpgradeFactory = nullptr;
74 
75  for (const std::string& searchPath : searchPaths) {
76  void* handle =
77  net::DynamicLoader::dlOpen((searchPath + "/libsnodec-" + upgradeContextName + ".so").c_str(), RTLD_LAZY | RTLD_GLOBAL);
78 
79  if (handle != nullptr) {
80  SocketContextUpgradeFactory* (*getSocketContextUpgradeFactory)() =
81  net::DynamicLoader::dlSym<SocketContextUpgradeFactory* (*) ()>(
82  handle,
83  upgradeContextName + (role == SocketContextUpgradeFactory::Role::SERVER ? "Server" : "Client") +
84  "ContextUpgradeFactory");
85 
86  if (getSocketContextUpgradeFactory != nullptr) {
87  socketContextUpgradeFactory = getSocketContextUpgradeFactory();
88 
89  if (socketContextUpgradeFactory != nullptr) {
90  if (add(socketContextUpgradeFactory, handle)) {
91  VLOG(0) << "SocketContextUpgradeFactory created successfull: " << socketContextUpgradeFactory->name();
92  } else {
93  VLOG(0) << "UpgradeSocketContext already existing. Not using: " << socketContextUpgradeFactory->name();
94  socketContextUpgradeFactory->destroy();
95  socketContextUpgradeFactory = nullptr;
96  net::DynamicLoader::dlCloseDelayed(handle);
97  }
98  break;
99  } else {
100  net::DynamicLoader::dlCloseDelayed(handle);
101  VLOG(0) << "SocketContextUpgradeFactory not created: " << upgradeContextName;
102  }
103  } else {
104  net::DynamicLoader::dlCloseDelayed(handle);
105  VLOG(0) << "Not a Plugin \"" << upgradeContextName;
106  }
107  } else {
108  VLOG(0) << "Error dlopen: " << net::DynamicLoader::dlError();
109  }
110  }
111 
112  return socketContextUpgradeFactory;
113  }

116  SocketContextUpgradeFactory*
117  SocketContextUpgradeFactorySelector<SocketContextUpgradeFactory>::select(const std::string& upgradeContextName) {
118  SocketContextUpgradeFactory* socketContextUpgradeFactory = nullptr;
119 
120  if (socketContextUpgradePlugins.contains(upgradeContextName)) {
121  socketContextUpgradeFactory = socketContextUpgradePlugins[upgradeContextName].socketContextUpgradeFactory;
122  } else if (linkedSocketContextUpgradePlugins.contains(upgradeContextName)) {
123  socketContextUpgradeFactory = linkedSocketContextUpgradePlugins[upgradeContextName]();
124  add(socketContextUpgradeFactory);
125  } else if (!onlyLinked) {
126  socketContextUpgradeFactory = load(upgradeContextName);
127  }
128
