49  int Config::init(const std::string& name, int argc, char* argv[]) {
50  this->name = name;
51  this->argc = argc;
52  this->argv = argv;
53 
54  app.description("Configuration file for application " + name);
55

58 
59  CLI::Option* logFileOpt = app.add_option("-l,--log-file", _logFile, "Log to file");
60  logFileOpt->type_name("[path]");
61  logFileOpt->configurable();
62 
63  CLI::Option* allHelpOpt = app.set_help_all_flag("--help-all", "Expand all help");
64  allHelpOpt->configurable(false);
65 
66  CLI::Option* daemonizeOpt = app.add_flag("-d,!-f,--daemonize,!--foreground", _daemonize, "Start application as daemon");
67  daemonizeOpt->configurable();
68 
69  CLI::Option* killDaemonOpt = app.add_flag("-k,--kill", _kill, "Start application as daemon");
70  killDaemonOpt->configurable(false);
71 
72  parse();
73 
74  if (app.count("--help") == 0) {
75  app.set_config("--config", std::string(CONFFILEPATH) + "/" + name + ".conf", "Read an config file", false);
76  parse();
77  }
78 
79  setRequired(_dumpConfig == false);
80 
81  return 0;
82  }
