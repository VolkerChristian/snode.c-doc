100  SSL* startSSL(SSL_CTX* ctx, const utils::Timeval& initTimeout, const utils::Timeval& shutdownTimeout) {
101  this->initTimeout = initTimeout;
102  this->shutdownTimeout = shutdownTimeout;
103  if (ctx != nullptr) {
104  ssl = SSL_new(ctx);
105 
106  if (ssl != nullptr) {
107  if (SSL_set_fd(ssl, Socket::getFd()) == 1) {
108  SocketReader::ssl = ssl;
109  SocketWriter::ssl = ssl;

111  SSL_free(ssl);
112  ssl = nullptr;

229  void doReadShutdown() override {
230  if (SSL_get_shutdown(ssl) == (SSL_SENT_SHUTDOWN | SSL_RECEIVED_SHUTDOWN)) {
231  VLOG(0) << "SSL_Shutdown COMPLETED: Close_notify sent and received";
232  SocketWriter::doWriteShutdown();
233  } else {
234  VLOG(0) << "SSL_Shutdown WAITING: Close_notify received but not send";
235  }
236  }

238  void doWriteShutdown() override {
239  if ((SSL_get_shutdown(ssl) & SSL_SENT_SHUTDOWN) == 0) {
240  doSSLShutdown(
241  [this]() -> void { // thus send one
242  if (SSL_get_shutdown(ssl) == (SSL_SENT_SHUTDOWN | SSL_RECEIVED_SHUTDOWN)) {
243  VLOG(0) << "SSL_Shutdown COMPLETED: Close_notify sent and received";
244  SocketWriter::doWriteShutdown();
245  } else {
246  VLOG(0) << "SSL_Shutdown WAITING: Close_notify sent but not received";
247  }
248  },
249  [this]() -> void {
250  LOG(WARNING) << "SSL_shutdown: Handshake timed out";
251  SocketWriter::disable();
252  },
253  [this](int sslErr) -> void {
254  ssl_log("SSL_shutdown: Handshake failed", sslErr);
255  SocketWriter::disable();
