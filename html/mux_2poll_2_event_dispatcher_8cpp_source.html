53  void PollFds::modAdd(core::DescriptorEventReceiver* eventReceiver, short event) {
54  int fd = eventReceiver->getRegisteredFd();
55 
56  std::unordered_map<int, PollFdIndex>::iterator itPollFdIndex = pollFdIndices.find(fd);
57 
58  if (itPollFdIndex == pollFdIndices.end()) {
59  pollfds[interestCount].events = event;
60  pollfds[interestCount].fd = fd;
61

84  void PollFds::modDel(core::DescriptorEventReceiver* eventReceiver, short event) {
85  int fd = eventReceiver->getRegisteredFd();
86 
87  std::unordered_map<int, PollFdIndex>::iterator itPollFdIndex = pollFdIndices.find(fd);
88 
89  PollFdIndex& pollFdIndex = itPollFdIndex->second;
90 
91  pollfds[pollFdIndex.index].events &= static_cast<short>(~event); // tilde promotes to int
92  pollFdIndex.events &= static_cast<short>(~event); // tilde promotes to int

115  void PollFds::compress() {
116  std::vector<pollfd>::iterator it = pollfds.begin();
117  std::vector<pollfd>::iterator rit = pollfds.begin() + static_cast<long>(pollfds.size()) - 1;
118 
119  while (it < rit) {
120  while (it != pollfds.end() && it->fd != -1) {
121  ++it;
122  }
123 
124  while (rit >= it && rit->fd == -1) {
125  --rit;
126  }
127 
128  while (it < rit && it->fd == -1 && rit->fd != -1) {
129  pollfd tPollFd = *it;
130  *it = *rit;
131  *rit = tPollFd;
132  ++it;
133  --rit;
134  }
135  }
136 
137  while (pollfds.size() > (interestCount * 2) + 1) {
138  pollfds.resize(pollfds.size() / 2);
139  }
140 
141  for (uint32_t i = 0; i < interestCount; i++) {
142  if (pollfds[i].fd >= 0) {
143  pollFdIndices[pollfds[i].fd].index = i;
144  }
145  }
146  }
