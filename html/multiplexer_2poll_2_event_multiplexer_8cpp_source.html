54  void PollFdsManager::muxAdd(core::DescriptorEventReceiver* eventReceiver, short event) {
55  int fd = eventReceiver->getRegisteredFd();
56 
57  if (!pollFdIndices.contains(fd)) {
58  pollfds[nextIndex].events = event;
59  pollfds[nextIndex].fd = fd;
60 
61  pollFdIndices[fd].index = nextIndex;
62  pollFdIndices[fd].events = event;
63 
64  ++nextIndex;
65 
66  if (nextIndex == pollfds.size()) {
67  pollfds.resize(pollfds.size() * 2, {-1, 0, 0});
68  pollFdIndices.reserve(pollfds.size());
69  }
70  } else {
71  PollFdIndex& pollFdIndex = pollFdIndices[fd];
72 
73  pollfds[pollFdIndex.index].events |= event;
74  pollFdIndex.events |= event;
75  }
76  }

78  void PollFdsManager::muxDel(int fd, short event) {
79  std::unordered_map<int, PollFdIndex>::iterator itPollFdIndex = pollFdIndices.find(fd);
80 
81  PollFdIndex& pollFdIndex = itPollFdIndex->second;
82 
83  pollfds[pollFdIndex.index].events &= static_cast<short>(~event); // tilde promotes to int
84  pollFdIndex.events &= static_cast<short>(~event); // tilde promotes to int
85 
86  if (pollFdIndex.events == 0) {
87  pollfds[pollFdIndex.index].fd = -1; // Compress will keep track of that descriptor
88  pollFdIndices.erase(fd);
89 
90  if (pollfds.size() > (pollFdIndices.size() * 2) + 1) {
91  compress();
92  }
93  }
94  }

106  void PollFdsManager::compress() {
107  remove_if(pollfds.begin(), pollfds.end(), [](const pollfd& pollFd) -> bool {
108  return pollFd.fd < 0;
109  });
110 
111  pollfds.resize(pollFdIndices.size() + 1, {-1, 0, 0});
112 
113  pollFdIndices.reserve(pollFdIndices.size() + 1);
114 
115  for (uint32_t i = 0; i < pollFdIndices.size(); i++) {
116  if (pollfds[i].fd >= 0) {
117  pollFdIndices[pollfds[i].fd].index = i;
118  }
119  }
120 
121  nextIndex = pollFdIndices.size();
122  }
